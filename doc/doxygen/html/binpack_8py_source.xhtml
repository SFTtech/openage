11 def factor(n):
12  """Return two (preferable close) factors of n."""
13  for a in range(int(math.sqrt(n)), 0, -1):
14  if n % a == 0:
15  return a, n // a

45 class RowPacker(Packer):
46  """
47  Packs blocks into rows, greedily trying to minimize the maximum width.
48  """
49 
50  def pack(self, blocks):
51  self.mapping = {}
52 
53  num_rows, _ = factor(len(blocks))
54  rows = [[] for _ in range(num_rows)]
55 
56  # Put blocks into rows.
57  for block in blocks:
58  min_row = min(rows, key=lambda row: sum(b.width for b in row))
59  min_row.append(block)
60

62  y = 0
63  for row in rows:
64  x = 0
65 
66  for block in row:
67  self.mapping[block] = (x, y)
68  x += block.width + self.margin

149 class BinaryTreePacker(Packer):
150  """
151  Binary tree bin packing strategy.
152 
153  Very close to http://codeincomplete.com/posts/2011/5/7/bin_packing/.
154 
155  Aditionally can target a given aspect ratio. 97/49 is optimal for terrain
156  textures.
157  """

195  def find_node(self, root, width, height):
196  if root.used:
197  return (self.find_node(root.right, width, height) or
198  self.find_node(root.down, width, height))
199  elif width <= root.width and height <= root.height:

202  def split_node(self, node, width, height):
203  node.used = True
204  node.down = PackerNode(node.x, node.y + height,
205  node.width, node.height - height)
206  node.right = PackerNode(node.x + width, node.y,
207  node.width - width, height)

210  def grow_node(self, width, height):
211  can_grow_down = width <= self.root.width
212  can_grow_right = height <= self.root.height
213  assert can_grow_down or can_grow_right, "Bad block ordering heuristic"
214 
215  should_grow_right = ((self.root.height * self.aspect_ratio) >=
216  (self.root.width + width))
217  should_grow_down = ((self.root.width / self.aspect_ratio) >=
218  (self.root.height + height))
219 
220  if can_grow_right and should_grow_right:
221  return self.grow_right(width, height)
222  elif can_grow_down and should_grow_down:
223  return self.grow_down(width, height)
224  elif can_grow_right:
225  return self.grow_right(width, height)
226  else:
227  return self.grow_down(width, height)
