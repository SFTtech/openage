16 # Linux-specific dirs according to the freedesktop basedir standard:
17 # https://standards.freedesktop.org/basedir-spec/basedir-spec-latest.html
18 #
19 # concretely:
20 # $XDG_CONFIG_HOME = $HOME/.config
21 # $XDG_DATA_HOME = $HOME/.local/share
22 # $XDG_DATA_DIRS = /usr/local/share/:/usr/share/
23 # $XDG_CONFIG_DIRS = /etc/xdg
24 # $XDG_CACHE_HOME = $HOME/.cache
25 # $XDG_RUNTIME_DIR = /run/user/$UID
26 #
27 LINUX_DIRS = {
28  "config_home": ("XDG_CONFIG_HOME", ("{HOME}/.config", {"HOME"})),
29  "data_home": ("XDG_DATA_HOME", ("{HOME}/.local/share", {"HOME"})),
30  "data_dirs": ("XDG_DATA_DIRS", ("/usr/local/share/:/usr/share/", {})),
31  "config_dirs": ("XDG_CONFIG_DIRS", ("/etc/xdg", {})),
32  "cache_home": ("XDG_CACHE_HOME", ("{HOME}/.cache", {"HOME"})),
33  "runtime_dir": ("XDG_RUNTIME_DIR", ("/run/user/$UID")),
34 }
35 
36 
37 # Windows-specific paths
38 WINDOWS_DIRS = {
39  "config_home": ("APPDATA", (False, None)),
40  "data_home": ("APPDATA", (False, None)),
41  "config_dirs": ("ALLUSERSPROFILE", (False, None)),
42  # TODO: other windows paths
43 }

46 def get_dir(which):
47  """
48  Returns directories used for data and config storage.
49  returns pathlib.Path
50  """
51 
52  platform_table = None
53 
54  if sys.platform.startswith("linux"):
55  platform_table = LINUX_DIRS
56 
57  elif sys.platform.startswith("darwin"):
58  raise Exception("macOS not really supported")
59 
60  elif sys.platform.startswith("win32"):
61  platform_table = WINDOWS_DIRS
62 
63  else:
64  raise Exception("unsupported platform: '%s'" % sys.platform)
65 
66  if which not in platform_table:
67  raise ValueError("unknown directory requested: '{}'".format(which))
68 
69  # fetch the directory template
70  env_var, (default_template, required_envs) = platform_table[which]
71 
72  # then create the result from the environment
73  env_val = os.environ.get(env_var)
74 
75  if env_val:
76  path = env_val
77 
78  elif default_template:
79  env_vars = {var: os.environ.get(var) for var in required_envs}
80  if not all(env_vars.values()):
81  raise Exception("could not reconstruct {}, "
82  "missing env variables: '{}'".format(
83  which,
84  [var for (var, val) in env_var.items()
85  if val is None].join(", ")
86  ))
87 
88  path = default_template.format(**env_vars)
89 
90  else:
91  raise Exception("could not find '{}' in environment".format(which))
92 
93  return pathlib.Path(path)
