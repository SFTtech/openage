132 bool SoundImpl::mix_audio(int32_t *stream, int length) {
133  size_t stream_index = 0;
134  while (length > 0) {
135  // fetch the raw audio from the underlying resource
136  auto chunk = resource->get_data(offset, length);
137 
138  if (chunk.length == 0) {
139  if (this->looping) {
140  offset = 0;
141  } else {
142  this->playing = false;
143  return true;
144  }
145  } else if (chunk.data == nullptr) {
146  return false;
147  }
148 
149  for (size_t i = 0; i < chunk.length; i++) {
150  stream[i+stream_index] += this->volume * chunk.data[i];
151  }
152 
153  this->offset += chunk.length;
154  length -= chunk.length;
155  stream_index += chunk.length;
156  }
157 
158  return false;
159 }
