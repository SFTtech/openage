158  def read_directory(self):
159  """
160  reads the directory that's currently pointed at by self.data.
161 
162  descends recursively for subdirectories.
163 
164  returns a ResourceDirectory object with both subdirs and leaves filled
165  in.
166  """
167  directory = ResourceDirectory.read(self.data)
168 
169  result = {}
170 
171  # read directory entries
172  entry_count = directory.named_entry_count + directory.id_entry_count
173  for idx in range(entry_count):
174  entry = ResourceDirectoryEntry.read(self.data)
175 
176  # check the entry name
177  entry.name_is_str = bool(entry.name & (1 << 31))
178 
179  if entry.name_is_str and idx >= directory.named_entry_count:
180  raise Exception("expected an id entry, but got a str entry")
181 
182  if not entry.name_is_str and idx < directory.named_entry_count:
183  raise Exception("expected a str entry, but got an id entry")
184 
185  # read the entry name string, if needed
186  if entry.name_is_str:
187  data_pos = self.data.tell()
188  self.data.seek(entry.name - (1 << 31))
189  entry.name = StringLiteral.readall(self.data).value
190  self.data.seek(data_pos)
191 
192  # read the struct pointed at by entry.data
193  entry.is_subdir = bool(entry.data & (1 << 31))
194  data_pos = self.data.tell()
195  if entry.is_subdir:
196  # recursively descend
197  self.data.seek(entry.data - (1 << 31))
198  result[entry.name] = self.read_directory()
199  else:
200  # read leaf metadata
201  self.data.seek(entry.data)
202  leaf = ResourceLeaf.read(self.data)
203  leaf.data_ptr -= self.datava
204  leaf.fileobj = self.data
205  result[entry.name] = leaf
206  self.data.seek(data_pos)
207 
208  return result

210  def read_strings(self):
211  """
212  reads all ressource strings from self.root;
213  returns a dict of dicts: {languageid: {stringid: str}}
214  """
215  result = defaultdict(lambda: {})
216 
217  try:
218  string_dir = self.rootdir[RESOURCE_IDS['string']]
219  except KeyError:
220  return result
221 
222  # strings are grouped in tables of 16 each.
223  for table_id, table_dir in string_dir.items():
224  # counting from 1. great idea, Microsoft.
225  base_string_id = (table_id - 1) * STRINGTABLE_SIZE
226 
227  # each table has leafs for one or more languages
228  for lang_id, string_table in table_dir.items():
229  langcode = LANGCODES[lang_id]
230  string_table_resource = string_table.open()
231  for idx in range(STRINGTABLE_SIZE):
232  string = StringLiteral.readall(string_table_resource).value
233  if string:
234  result[langcode][base_string_id + idx] = string
235 
236  if sum(string_table_resource.read()) > 0:
237  raise Exception("string table invalid: the padding "
238  "contains data.")
239 
240  return result
