193 template <typename T>
194 typename KeyframeContainer<T>::KeyframeIterator KeyframeContainer<T>::last(const time_t &time,
195  const KeyframeIterator &hint) const {
196  KeyframeIterator e = (hint == this->container.end()) ? this->container.begin() : hint;
197 
198  if (this->container.front().time > time) {
199  // This will never happen due to the container.front->time == -Inf magic!
200  throw Error(ERR << "rupture in spacetime detected, curve container is broken");
201  }
202 
203  // Search in the queue
204  if (time > e->time) { // the searched element is supposed to be AFTER the hint
205  // perform the search via ->next
206  while (e != this->container.end() && time >= e->time) {
207  e++;
208  }
209  e--;
210  // e is now one of two options:
211  // 1. e == end: The last element of the queue was smaller than `time`
212  // 2. e != end: There was a element with `e->time` > `time`
213 
214  } else if (time < e->time) {
215  // the searched element is supposed to be BEFORE the hint
216  // perform the search via ->prev
217  while (e != this->container.begin() && time < e->time) {
218  e--;
219  }
220  // e is now one of two options:
221  // 1. e == begin: The time was before every element in the queue
222  // 2. e != begin: There was an element with `e->time` > `time`
223 
224  } else {
225  // perform <= search - and return e, whose time is == time.
226  }
227 
228  return e;
229 }
