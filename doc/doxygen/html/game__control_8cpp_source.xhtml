69 void ActionModeSignals::on_action(const std::string &action_name) {
70  Engine *engine = this->action_mode->game_control->get_engine();
71 
72  const input::action_t &action = engine->get_action_manager().get(action_name);
73  input::InputContext *top_ctxt = &engine->get_input_manager().get_top_context();
74 
75  if (top_ctxt == this->action_mode ||
76  top_ctxt == &this->action_mode->building_context ||
77  top_ctxt == &this->action_mode->build_menu_context ||
78  top_ctxt == &this->action_mode->build_menu_mil_context) {
79 
80  // create an action that just relays the action.
81  input::action_arg_t action_arg{
82  input::Event{input::event_class::ANY, 0, input::modset_t{}},
83  coord::input{0, 0},
84  coord::input_delta{0, 0},
85  {action}
86  };
87  top_ctxt->execute_if_bound(action_arg);
88  }

411 bool ActionMode::place_selection(coord::phys3 point) {
412  if (this->type_focus) {
413  auto player = this->game_control->get_current_player();
414 
415  if (player->can_make(*this->type_focus)) {
416 
417  // confirm building placement with left click
418  // first create foundation using the producer
419  Engine *engine = this->game_control->get_engine();
420 
421  UnitContainer *container = &engine->get_game()->placed_units;
422  UnitReference new_building = container->new_unit(*this->type_focus, *player, point);
423 
424  // task all selected villagers to build
425  // TODO: editor placed objects are completed already
426  if (new_building.is_valid()) {
427 
428  player->deduct(this->type_focus->cost.get(*player)); // TODO change, move elsewheres
429 
430  Command cmd(*player, new_building.get());
431  cmd.set_ability(ability_type::build);
432  cmd.add_flag(command_flag::interrupt);
433  this->selection->all_invoke(cmd);
434  return true;
435  }
436 
437  } else {
438  // TODO show in game error message

440  }
441  }

486 void ActionMode::announce_resources() {
487  if (this->game_control) {
488  if (Player *player = this->game_control->get_current_player()) {
489  for (auto i = static_cast<std::underlying_type<game_resource>::type>(game_resource::RESOURCE_TYPE_COUNT); i != 0; --i) {
490  auto resource_type = static_cast<game_resource>(i - 1);
491 
492  emit this->gui_signals.resource_changed(
493  resource_type,

496  }
497  emit this->gui_signals.population_changed(player->population.get_demand(), player->population.get_capacity(),
498  player->population.get_space() <= 0 && !player->population.is_capacity_maxed());
499  }

503 void ActionMode::announce_buttons_type() {
504  ActionButtonsType buttons_type;
505  Engine *engine = this->game_control->get_engine();
506  InputContext *top_ctxt = &engine->get_input_manager().get_top_context();
507  if (top_ctxt == &this->build_menu_context) {
508  buttons_type = ActionButtonsType::BuildMenu;
509  } else if (top_ctxt == &this->build_menu_mil_context) {
510  buttons_type = ActionButtonsType::MilBuildMenu;
511  } else if (top_ctxt == &this->building_context ||
512  this->selection->get_selection_type() != selection_type_t::own_units) {
513  buttons_type = ActionButtonsType::None;
514  } else if (this->selection->contains_military(*this->game_control->get_current_player())) {
515  buttons_type = ActionButtonsType::MilitaryUnits;
516  } else {
517  buttons_type = ActionButtonsType::CivilianUnits;
518  }
519 
520  if (buttons_type != this->buttons_type) {
521  this->buttons_type = buttons_type;
522  emit this->gui_signals.buttons_type_changed(buttons_type);
523 
524  // announce the changed input context
525  this->announce();
526  }

856 void GameControl::set_mode(int mode_index, bool signal_if_unchanged) {
857  bool need_to_signal = signal_if_unchanged;
858 
859  // do we wanna set a new mode?
860  if (mode_index != -1) {
861 
862  // if the new mode is valid, available and not active at the moment
863  if (mode_index < std::distance(std::begin(this->modes),
864  std::end(this->modes))
865  && this->modes[mode_index]->available()
866  && this->active_mode_index != mode_index) {
867 
868  ENSURE(!this->active_mode == (this->active_mode_index == -1),
869  "inconsistency between the active mode index and pointer");
870 
871  // exit from the old mode
872  if (this->active_mode) {
873  this->engine->get_input_manager().remove_context(
874  this->active_mode
875  );
876 
877  // trigger the exit callback of the mode
878  if (this->active_mode) {
879  this->active_mode->on_exit();
880  }
881  }
882 
883  // set the new active mode
884  this->active_mode_index = mode_index;
885  this->active_mode = this->modes[mode_index];
886 
887  // trigger the entry callback
888  this->active_mode->on_enter();
889 
890  // add the mode-local input context
891  this->engine->get_input_manager().push_context(
892  this->active_mode
893  );
894 
895  need_to_signal = true;
896  }
897  }
898  else {
899  // unassign the active mode
900  if (this->active_mode) {
901 
902  // remove the input context
903  this->engine->get_input_manager().remove_context(
904  this->active_mode
905  );
906 
907  this->active_mode_index = -1;
908  this->active_mode = nullptr;
909 
910  need_to_signal = true;
911  }
912  }
913 
914  if (need_to_signal) {
915  this->announce_mode();
916  }
917 }
