28 std::unordered_set<terrain_t> allowed_terrains(const gamedata::ground_type &restriction) {
29  // returns a terrain whitelist for a given restriction
30  // you can also define a blacklist for brevity, it will be converted and returned
31  std::unordered_set<terrain_t> whitelist;
32  std::unordered_set<terrain_t> blacklist;
33 
34  // 1, 14, and 15 are water, 2 is shore
35  if (restriction == gamedata::ground_type::WATER ||
36  restriction == gamedata::ground_type::WATER_0x0D ||
37  restriction == gamedata::ground_type::WATER_SHIP_0x03 ||
38  restriction == gamedata::ground_type::WATER_SHIP_0x0F) {
39 
40  whitelist.insert(1); // water
41  whitelist.insert(2); // shore
42  whitelist.insert(4); // shallows
43  whitelist.insert(14); // medium water
44  whitelist.insert(15); // deep water
45  }
46  else if (restriction == gamedata::ground_type::SOLID) {
47  blacklist.insert(1); // water
48  blacklist.insert(4); // shallows
49  blacklist.insert(14); // medium water
50  blacklist.insert(15); // deep water
51  }
52  else if (restriction == gamedata::ground_type::FOUNDATION ||
53  restriction == gamedata::ground_type::NO_ICE_0x08 ||
54  restriction == gamedata::ground_type::FOREST) {
55 
56  blacklist.insert(1); // water
57  blacklist.insert(4); // shallows
58  blacklist.insert(14); // medium water
59  blacklist.insert(15); // deep water
60  blacklist.insert(18); // ice
61  }
62  else {
63  log::log(MSG(warn) << "undefined terrain restriction, assuming solid");
64  blacklist.insert(1); // water
65  blacklist.insert(4); // shallows
66  blacklist.insert(14); // medium water
67  blacklist.insert(15); // deep water
68  }
69 
70  // if we're using a blacklist, fill out a whitelist with everything not on it
71  if (blacklist.size() > 0) {
72  // Allow all terrains that are not on blacklist
73  for (int i = 0; i < 32; ++i) {
74  if (blacklist.count(i) == 0) {
75  whitelist.insert(i);
76  }
77  }
78  }
79 
80  return whitelist;
81 }

251  }
252  else if (this->unit_data.unit_class == gamedata::unit_classes::GOLD_MINE) {
253  unit->add_attribute(std::make_shared<Attribute<attr_type::resource>>(game_resource::gold, 800));
254  }
255  else if (this->unit_data.unit_class == gamedata::unit_classes::STONE_MINE) {

273  if (t) {
274  t->initialise(unit, player);
275  }
276  }
277  ),
278  true);
279  }
280  else if (this->graphics.count(graphic_type::dying) > 0) {
281  unit->push_action(std::make_unique<DeadAction>(unit), true);
282  }
283 
284  // the default action
285  unit->push_action(std::make_unique<IdleAction>(unit), true);
286  }
287 
288  // give required abilitys
289  for (auto &a : this->type_abilities) {
290  unit->give_ability(a);
291  }
292 }

294 TerrainObject *ObjectProducer::place(Unit *u, std::shared_ptr<Terrain> terrain, coord::phys3 init_pos) const {
295 
296  // create new object with correct base shape
297  if (this->unit_data.selection_shape > 1) {
298  u->make_location<RadialObject>(this->unit_data.radius_x, this->terrain_outline);
299  }
300  else {
301  u->make_location<SquareObject>(this->foundation_size, this->terrain_outline);
302  }
303 
304  // find set of allowed terrains
305  std::unordered_set<terrain_t> terrains = allowed_terrains(this->unit_data.terrain_restriction);
306 
307  /*
308  * decide what terrain is passable using this lambda
309  * currently unit avoids water and tiles with another unit
310  * this function should be true if pos is a valid position of the object
311  */
312  TerrainObject *obj_ptr = u->location.get();
313  std::weak_ptr<Terrain> terrain_ptr = terrain;
314  u->location->passable = [obj_ptr, terrain_ptr, terrains](const coord::phys3 &pos) -> bool {
315 
316  // if location is deleted, then so is this lambda (deleting terrain implies location is deleted)
317  // so locking objects here will not return null
318  auto terrain = terrain_ptr.lock();
319 
320  // look at all tiles in the bases range
321  for (coord::tile check_pos : tile_list(obj_ptr->get_range(pos, *terrain))) {
322  TileContent *tc = terrain->get_data(check_pos);
323 
324  // invalid tile types
325  if (!tc || terrains.count(tc->terrain_id) == 0) {
326  return false;
327  }
328 
329  // compare with objects intersecting the units tile
330  // ensure no intersections with other objects
331  for (auto obj_cmp : tc->obj) {
332  if (obj_ptr != obj_cmp &&
333  obj_cmp->check_collisions() &&
334  obj_ptr->intersects(*obj_cmp, pos)) {
335  return false;
336  }
337  }
338  }
339  return true;
340  };
341 
342  u->location->draw = [u, obj_ptr](const Engine &e) {
343  if (u->selected) {
344  obj_ptr->draw_outline(e.coord);
345  }
346  u->draw(e);
347  };
348 
349  // try to place the obj, it knows best whether it will fit.
350  auto state = this->decay? object_state::placed_no_collision : object_state::placed;
351  if (u->location->place(terrain, init_pos, state)) {
352  if (this->on_create) {
353  this->on_create->play();
354  }
355  return u->location.get();
356  }
357 
358  // placing at the given position failed
359  u->log(MSG(dbg) << "failed to place object");
360  return nullptr;
361 }

682 TerrainObject *BuildingProducer::place(Unit *u, std::shared_ptr<Terrain> terrain, coord::phys3 init_pos) const {
683 
684  // buildings have a square base
685  u->make_location<SquareObject>(this->foundation_size, this->terrain_outline);
686 
687  /*
688  * decide what terrain is passable using this lambda
689  * currently unit avoids water and tiles with another unit
690  * this function should be true if pos is a valid position of the object
691  */
692  TerrainObject *obj_ptr = u->location.get();
693  std::weak_ptr<Terrain> terrain_ptr = terrain;
694 
695  // find set of allowed terrains
696  std::unordered_set<terrain_t> terrains = allowed_terrains(this->unit_data.terrain_restriction);
697 
698  u->location->passable = [obj_ptr, terrain_ptr, terrains](const coord::phys3 &pos) -> bool {
699  auto terrain = terrain_ptr.lock();
700 
701  // look at all tiles in the bases range
702  for (coord::tile check_pos : tile_list(obj_ptr->get_range(pos, *terrain))) {
703  TileContent *tc = terrain->get_data(check_pos);
704 
705  // check if terrains are suitable and free of content
706  if (!tc || !terrains.count(tc->terrain_id) || tc->obj.size()) {
707  return false;
708  }
709  }
710 
711  return true;
712  };
713 
714  // drawing function
715  bool draw_outline = this->enable_collisions;
716  u->location->draw = [u, obj_ptr, draw_outline](const Engine &e) {
717  if (u->selected && draw_outline) {
718  obj_ptr->draw_outline(e.coord);
719  }
720  u->draw(e);
721  };
722 
723  // try to place the obj, it knows best whether it will fit.
724  auto state = object_state::floating;
725  if (!u->location->place(terrain, init_pos, state)) {
726  return nullptr;
727  }
728 
729  // annex objects
730  for (unsigned i = 0; i < 4; ++i) {
731  const gamedata::building_annex &annex = this->unit_data.building_annex.data[i];
732  if (annex.unit_id > 0) {
733 
734  // make objects for annex
735  coord::phys3 a_pos = u->location->pos.draw;
736  a_pos.ne += annex.misplaced0;
737  a_pos.se += annex.misplaced1;
738  this->make_annex(*u, terrain, annex.unit_id, a_pos, i == 0);
739  }
740  }
741 
742  // TODO: play sound once built
743  if (this->on_create) {
744  this->on_create->play();
745  }
746  return u->location.get();
747 }

749 TerrainObject *BuildingProducer::make_annex(Unit &u, std::shared_ptr<Terrain> t,
750  int annex_id, coord::phys3 annex_pos, bool c) const {
751 
752  // for use in lambda drawing functions
753  auto annex_type = this->owner.get_type(annex_id);
754  if (!annex_type) {
755  u.log(MSG(warn) << "Invalid annex type id " << annex_id);
756  return nullptr;
757  }
758 
759  // foundation size
760  coord::tile_delta annex_foundation = annex_type->foundation_size;
761 
762  // producers place by the nw tile
763  coord::phys3 start_tile = annex_pos;
764  start_tile.ne -= annex_foundation.ne / 2.0;
765  start_tile.se -= annex_foundation.se / 2.0;
766 
767  // create and place on terrain
768  TerrainObject *annex_loc = u.location->make_annex<SquareObject>(annex_foundation);
769  object_state state = c? object_state::placed : object_state::placed_no_collision;
770  annex_loc->place(t, start_tile, state);
771 
772  // create special drawing functions for annexes,
773  annex_loc->draw = [annex_loc, annex_type, &u, c](const Engine &e) {
774 
775  // hack which draws the outline in the right order
776  // removable once rendering system is improved
777  if (c && u.selected) {
778  annex_loc->get_parent()->draw_outline(e.coord);
779  }
780 
781  // only draw if building is completed
782  if (u.has_attribute(attr_type::building) &&
783  u.get_attribute<attr_type::building>().completed >= 1.0f) {
784  u.draw(annex_loc, annex_type->graphics, e);
785  }
786  };
787  return annex_loc;

847 TerrainObject *ProjectileProducer::place(Unit *u, std::shared_ptr<Terrain> terrain, coord::phys3 init_pos) const {
848  /*
849  * radial base shape without collision checking
850  */
851  u->make_location<RadialObject>(this->unit_data.radius_y, this->terrain_outline);
852 
853  TerrainObject *obj_ptr = u->location.get();
854  std::weak_ptr<Terrain> terrain_ptr = terrain;
855  u->location->passable = [obj_ptr, u, terrain_ptr](const coord::phys3 &pos) -> bool {
856  if (pos.up > 64000) {
857  return true;
858  }
859 
860  // avoid intersections with launcher
861  Unit *launcher = nullptr;
862  auto terrain = terrain_ptr.lock();
863  if (u->has_attribute(attr_type::projectile)) {
864  auto &pr_attr = u->get_attribute<attr_type::projectile>();
865  if (pr_attr.launched && pr_attr.launcher.is_valid()) {
866  launcher = pr_attr.launcher.get();
867  }
868  else {
869  return true;
870  }
871  }
872  else {
873  return true;
874  }
875 
876  // look at all tiles in the bases range
877  for (coord::tile check_pos : tile_list(obj_ptr->get_range(pos, *terrain))) {
878  TileContent *tc = terrain->get_data(check_pos);
879  if (!tc) return false;
880 
881  // ensure no intersections with other objects
882  for (auto obj_cmp : tc->obj) {
883  if (obj_ptr != obj_cmp &&
884  &obj_cmp->unit != launcher &&
885  obj_cmp->check_collisions() &&
886  obj_ptr->intersects(*obj_cmp, pos)) {
887  return false;
888  }
889  }
890  }
891  return true;
892  };
893 
894  u->location->draw = [u](const Engine &e) {
895  u->draw(e);
896  };
897 
898  // try to place the obj, it knows best whether it will fit.
899  if (u->location->place(terrain, init_pos, object_state::placed_no_collision)) {
900  return u->location.get();
901  }
902  return nullptr;
903 }
