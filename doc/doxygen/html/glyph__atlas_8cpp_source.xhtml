100 GlyphAtlas::Entry GlyphAtlas::set(size_t key, const Glyph &glyph, const unsigned char *image) {
101  // Give the glyphs a 1px padding in the atlas
102  int required_width = glyph.width + 1;
103  int required_height = glyph.height + 1;
104 
105  if (this->shelves.empty()) {
106  this->shelves.emplace_back(0, this->width, required_height);
107  }
108 
109  for (auto &shelf : this->shelves) {
110  if (shelf.check_fits(required_width, required_height)) {
111  int x_pos = shelf.reserve(required_width, required_height);
112  int y_pos = shelf.y_position;
113 
114  // Create a new entry and insert it
115  GlyphAtlas::Entry entry;
116  entry.glyph = glyph;
117  entry.u0 = static_cast<float>(x_pos)/this->width;
118  entry.v0 = static_cast<float>(y_pos)/this->height;
119  entry.u1 = static_cast<float>(x_pos + glyph.width)/this->width;
120  entry.v1 = static_cast<float>(y_pos + glyph.height)/this->height;
121 
122  for (unsigned int i = 0; i < glyph.height; i++) {
123  memcpy(
124  this->buffer.get() + ((y_pos + i) * this->width + x_pos),
125  image + (i * glyph.width),
126  glyph.width * sizeof(unsigned char)
127  );
128  }
129  this->update_dirty_area(x_pos, y_pos, glyph.width, glyph.height);
130 
131  this->glyphs.emplace(key, entry);
132  return entry;
133  }
134  }
135 
136  GlyphAtlas::Shelf last_shelf = this->shelves.back();
137  if (this->height > (last_shelf.y_position + last_shelf.height + required_height)) {
138  this->shelves.emplace_back(last_shelf.y_position + last_shelf.height, this->width, required_height);
139 
140  return this->set(key, glyph, image);
141  }
142 
143  // TODO: figure out a clean way to resize the atlas
144  throw Error(MSG(err) << "Not enough space in the atlas");
145 }
