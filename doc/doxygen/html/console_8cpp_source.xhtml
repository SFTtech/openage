126 void Console::interpret(const std::string &command) {
127  if (command == "exit") {
128  this->set_visible(false);
129  }
130  else if (command == "list") {
131  for (auto &line : this->engine->list_options()) {
132  this->write(line.c_str());
133  }
134  }
135  else if (command.substr(0,3) == "set") {
136  std::size_t first_space = command.find(" ");
137  std::size_t second_space = command.find(" ", first_space+1);
138  if (first_space != std::string::npos && second_space != std::string::npos) {
139  std::string name = command.substr(first_space+1, second_space-first_space-1);
140  std::string value = command.substr(second_space+1, std::string::npos);
141  this->engine->get_cvar_manager().set(name,value);
142  }
143  }
144  else if (command.substr(0,3) == "get") {
145  std::size_t first_space = command.find(" ");
146  if (first_space != std::string::npos) {
147  std::string name = command.substr(first_space+1, std::string::npos);
148  std::string value = this->engine->get_cvar_manager().get(name);
149  this->write(value.c_str());
150  }
151  }
152 }
