14 class Union(FSLikeObject):
15  """
16  FSLikeObject that provides a structure for mounting several path objects.
17 
18  Unlike in POSIX, mounts may overlap.
19  If multiple mounts match for a directory, those that have a higher
20  priority are preferred.
21  In case of equal priorities, later mounts are preferred.
22  """

48  def add_mount(self, pathobj, mountpoint, priority):
49  """
50  This method should not be called directly; instead, use the mount
51  method of Path objects that were obtained from this.
52 
53  Mounts pathobj at mountpoint, with the given priority.
54  """
55 
56  if not isinstance(pathobj, Path):
57  raise Exception("only a fslike.Path can be mounted, "
58  "not {}".format(type(pathobj)))
59 
60  # search for the right place to insert the mount.
61  idx = len(self.mounts) - 1
62  while idx >= 0 and priority >= self.mounts[idx][2]:
63  idx -= 1
64 
65  self.mounts.insert(idx + 1, (tuple(mountpoint), pathobj, priority))
66 
67  # 'create' parent directories as needed.
68  dirstructure = self.dirstructure
69  for subdir in mountpoint:
70  dirstructure = dirstructure.setdefault(subdir, {})

72  def remove_mount(self, search_mountpoint, source_pathobj=None):
73  """
74  Remove a mount from the union by searching for the source
75  that provides the given mountpoint.
76  Additionally, can check if the source equals the given pathobj.
77  """
78 
79  unmount = []
80 
81  for idx, (mountpoint, pathobj, _) in enumerate(self.mounts):
82  # cut the search so prefixes can be matched.
83  if mountpoint == tuple(search_mountpoint[:len(mountpoint)]):
84  if not source_pathobj or source_pathobj == pathobj:
85  unmount.append(idx)
86 
87  if unmount:
88  # reverse the order so that the indices never shift.
89  for idx in reversed(sorted(unmount)):
90  del self.mounts[idx]

93  raise ValueError("could not find mounted source")
