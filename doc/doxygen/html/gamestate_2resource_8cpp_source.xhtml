49 void ResourceBundle::expand(const int count) {
50  if (this->count == count) {
51  return;
52  }
53  // create new array with old values
54  double *new_value = new double[count];
55  for (int i = 0; i < this->count; i++) {
56  new_value[i] = this->value[i];
57  }
58  // replace the private variables
59  this->count = count;
60  delete[] value;
61  this->value = new_value;
62 }
