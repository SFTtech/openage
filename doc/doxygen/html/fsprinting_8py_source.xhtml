68 def print_tree(obj, path="", prefix="", max_entries=INF):
69  """
70  Obj is a filesystem-like object; path must be a string.
71 
72  Recursively descends into subdirectories using prefix.
73 
74  If max_entries is given, only that number of entries per directory
75  is printed.
76  """
77  # entries is a list of tuples of entryname, isdir, ismeta
78  entries = []
79  for entry in obj.listdirs(path):
80  entries.append((entry, True, False))

82  entries.append((entry, False, False))

86 
87  if len(entries) > max_entries:
88  omit = len(entries) - max_entries + 1
89  entries = entries[:-omit] + [("[%d omitted]" % omit, False, True)]
