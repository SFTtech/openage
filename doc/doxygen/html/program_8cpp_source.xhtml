72 void Program::check(GLenum what_to_check) {
73  GLint status;
74  glGetProgramiv(this->id, what_to_check, &status);
75 
76  if (status != GL_TRUE) {
77  GLint loglen;
78  glGetProgramiv(this->id, GL_INFO_LOG_LENGTH, &loglen);
79  char *infolog = new char[loglen];
80  glGetProgramInfoLog(this->id, loglen, NULL, infolog);
81 
82  const char *what_str;
83  switch(what_to_check) {
84  case GL_LINK_STATUS:
85  what_str = "linking";
86  break;
87  case GL_VALIDATE_STATUS:
88  what_str = "validation";
89  break;
90  case GL_COMPILE_STATUS:
91  what_str = "compiliation";
92  break;
93  default:
94  what_str = "<unknown task>";
95  break;
96  }
97 
98  auto errormsg = MSG(err);
99  errormsg << "Program " << what_str << " failed\n" << infolog;
100  delete[] infolog;

103  }
