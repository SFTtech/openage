140 void Profiler::end_frame_measure() {
141  if (not this->engine_in_debug_mode()) {
142  return;
143  }
144 
145  auto frame_end = std::chrono::high_resolution_clock::now();
146  this->frame_duration = frame_end - this->frame_start;
147 
148  for (auto com : this->registered_components()) {
149  double percentage = this->duration_to_percentage(this->components[com].duration);

174  this->engine->render_text(position, 12, "%s", com.second.display_name.c_str());
175 
176  offset += PROFILER_COM_BOX_HEIGHT + 2;

180 double Profiler::duration_to_percentage(std::chrono::high_resolution_clock::duration duration) {
181  double dur = std::chrono::duration_cast<std::chrono::microseconds>(duration).count();
182  double ref = std::chrono::duration_cast<std::chrono::microseconds>(this->frame_duration).count();
183  double percentage = dur / ref * 100;
184  return percentage;
185 }
