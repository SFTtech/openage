107 def main(args, error):
108  """ Codegen CLI entry point. """
109 
110  if args.file_to_touch and not os.path.isfile(args.file_to_touch):
111  error("file doesn't exist: " + args.file_to_touch)
112 
113  def read_cache_file_lines(filename):
114  """
115  Yields all lines from the file.
116  If the file doesn't exist, a warning is printed.
117  """
118  try:
119  with open(filename) as fileobj:
120  for line in fileobj:
121  line = line.strip()
122  if line:
123  yield line
124  except FileNotFoundError:
125  if args.file_to_touch or args.force_rerun_on_generated_list_change:
126  print("warning: cache actions were requested, " +
127  "but the target cache could not be read!")
128 
129  old_generated = set(read_cache_file_lines(args.generated_list_file))
130  old_depends = set(read_cache_file_lines(args.depend_list_file))
131 
132  try:
133  mode = CodegenMode(args.mode)
134  except ValueError as exc:
135  error(exc.args[0])
136 
137  # arguments are OK.
138 
139  # generate sources
140  generated, depends = codegen(mode, args.input_dir, args.output_dir)
141 
142  def print_set_differences(old, new, name):
143  """ Prints the difference between old and new. """
144  if old - new:
145  print("codegen: removed from " + name + ":\n"
146  "\t" + "\n\t".join(old - new) + "\n")
147 
148  if new - old and old:
149  print("codegen: new " + name + ":\n"
150  "\t" + "\n\t".join(new - old) + "\n")
151 
152  print_set_differences(old_generated, generated, "generated files")
153  print_set_differences(old_depends, depends, "dependencies")
154 
155  with open(args.generated_list_file, 'w') as fileobj:
156  fileobj.write('\n'.join(generated))
157 
158  with open(args.depend_list_file, 'w') as fileobj:
159  fileobj.write('\n'.join(depends))
160 
161  if args.file_to_touch:
162  if old_generated != generated or old_depends != depends:
163  os.utime(args.file_to_touch)
164 
165  if args.force_rerun_on_generated_list_change:
166  if old_generated != generated:
167  print("\n\n\n\n"
168  "The list of generated source files has changed.\n"
169  "A build update has been triggered; you need to build "
170  "again.\n"
171  "\n\n\n")
172 
173  # fail
174  sys.exit(1)
