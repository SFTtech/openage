39 static vlk_capabilities find_capabilities() {
40  vlk_capabilities caps;
41 
42  // Find which layers are available.
43  auto layers = vk_do_ritual(vkEnumerateInstanceLayerProperties);
44 
45  log::log(MSG(dbg) << "Available Vulkan layers:");
46  for (auto const& lr : layers) {
47  caps.layers.insert(lr.layerName);
48  log::log(MSG(dbg) << "\t" << lr.layerName);
49  }
50 
51  // Find which extensions are available.
52  // This is annoying, since an enumeration call without a filter-by-layer parameter
53  // won't return all extensions. We thus have to enumerate extensions for each layer
54  // and then remove duplicates.
55 
56  // First retrieve non-layer extensions.
57  auto props = vk_do_ritual(vkEnumerateInstanceExtensionProperties, nullptr);
58 
59  for (auto const& p : props) {
60  caps.extensions.emplace(p.extensionName);
61  }
62 
63  // Then retrieve extensions from layers.
64  for (auto const& lr : layers) {
65  auto lr_props = vk_do_ritual(vkEnumerateInstanceExtensionProperties, lr.layerName);
66 
67  for (auto const& p : lr_props) {
68  caps.extensions.emplace(p.extensionName);
69  }
70  }
71 
72  log::log(MSG(dbg) << "Available Vulkan extensions:");
73  for (const auto& ext : caps.extensions) {
74  log::log(MSG(dbg) << "\t" << ext);
75  }
76 
77  return caps;
78 }

80 VlkWindow::VlkWindow(const char* title)
81  : Window(800, 600)
82  , capabilities(find_capabilities())
83 {
84  make_sdl_available();
85 
86  this->window = SDL_CreateWindow(
87  title,
88  SDL_WINDOWPOS_CENTERED,
89  SDL_WINDOWPOS_CENTERED,
90  this->size.first,
91  this->size.second,
92  SDL_WINDOW_OPENGL | SDL_WINDOW_RESIZABLE | SDL_WINDOW_MAXIMIZED
93  );
94 
95  if (this->window == nullptr) {
96  throw Error{MSG(err) << "Failed to create SDL window: " << SDL_GetError()};
97  }
98 
99  // Find which extensions the SDL window requires.
100  auto extension_names = vk_do_ritual(SDL_Vulkan_GetInstanceExtensions, this->window);
101  /*
102  if (succ != SDL_TRUE) {
103  throw Error(MSG(err) << "Failed to obtain required Vulkan extension names from SDL.");
104  }
105  */
106 
107 #ifndef NDEBUG
108  extension_names.push_back(VK_EXT_DEBUG_REPORT_EXTENSION_NAME);
109 #endif
110  extension_names.push_back(VK_KHR_SURFACE_EXTENSION_NAME);
111 
112  // Check if all extensions are available.
113  for (auto ext : extension_names) {
114  if (this->capabilities.extensions.count(ext) == 0) {
115  throw Error(MSG(err) << "Vulkan driver is missing required extension: " << ext);
116  }
117  }
118 
119  // The names of Vulkan layers which we need.
120  std::vector<const char*> layer_names;
121 #ifndef NDEBUG
122  layer_names.push_back("VK_LAYER_LUNARG_standard_validation");
123 #endif
124 
125  // Check if all layers are available
126  for (auto lr : layer_names) {
127  if (this->capabilities.layers.count(lr) == 0) {
128  throw Error(MSG(err) << "Vulkan driver is missing required layer: " << lr);
129  }
130  }
131 
132  // Setup application description.
133  VkApplicationInfo app_info = {};
134  app_info.sType = VK_STRUCTURE_TYPE_APPLICATION_INFO;
135  app_info.pApplicationName = title;
136  app_info.apiVersion = VK_MAKE_VERSION(1, 0, 57);
137 
138  VkInstanceCreateInfo inst_info = {};
139  inst_info.sType = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO;
140  inst_info.pApplicationInfo = &app_info;
141  inst_info.enabledExtensionCount = extension_names.size();
142  inst_info.ppEnabledExtensionNames = extension_names.data();
143  inst_info.enabledLayerCount = layer_names.size();
144  inst_info.ppEnabledLayerNames = layer_names.data();
145 
146  // A Vulkan instance is a proxy for all usage of Vulkan from our application,
147  // kind of like a GL context. Initialize it.
148  VK_CALL_CHECKED(vkCreateInstance, &inst_info, nullptr, &this->instance);
149 
150  this->loader.init(this->instance);
151 
152 #ifndef NDEBUG
153  VkDebugReportCallbackCreateInfoEXT cb_info = {};
154  cb_info.sType = VK_STRUCTURE_TYPE_DEBUG_REPORT_CALLBACK_CREATE_INFO_EXT;
155  cb_info.flags =
156  VK_DEBUG_REPORT_ERROR_BIT_EXT
157  | VK_DEBUG_REPORT_WARNING_BIT_EXT
158  | VK_DEBUG_REPORT_PERFORMANCE_WARNING_BIT_EXT
159  | VK_DEBUG_REPORT_INFORMATION_BIT_EXT
160  | VK_DEBUG_REPORT_DEBUG_BIT_EXT;
161  cb_info.pfnCallback = vlk_debug_cb;
162 
163  VK_CALL_CHECKED(this->loader.vkCreateDebugReportCallbackEXT, this->instance, &cb_info, nullptr, &this->debug_callback);
164 #endif
165 
166  // Surface is an object that we draw into, corresponding to the window area.
167  auto succ = SDL_Vulkan_CreateSurface(this->window, this->instance, &this->surface);
168  if (succ != SDL_TRUE) {
169  throw Error(MSG(err) << "Failed to create Vulkan surface on SDL window.");
170  }
171 }
