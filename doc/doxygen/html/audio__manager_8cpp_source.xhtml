48 AudioManager::AudioManager(job::JobManager *job_manager,
49  const std::string &device_name)
50  :
51  available{false},
52  job_manager{job_manager},
53  device_name{device_name} {
54 
55  if (SDL_Init(SDL_INIT_AUDIO) < 0) {
56  log::log(MSG(err)
57  << "SDL audio initialization failed: "
58  << SDL_GetError());
59  return;
60  } else {
61  log::log(MSG(info) << "SDL audio subsystems initialized");
62  }
63 
64  // set desired audio output format
65  SDL_AudioSpec desired_spec;
66  SDL_zero(desired_spec);
67  desired_spec.freq = 48000;
68  desired_spec.format = AUDIO_S16LSB;
69  desired_spec.channels = 2;
70  desired_spec.samples = 4096;
71  desired_spec.userdata = this;
72 
73  // call back that is invoked once SDL needs the next chunk of data
74  desired_spec.callback = [] (void *userdata, uint8_t *stream, int len) {
75  AudioManager *audio_manager = static_cast<AudioManager*>(userdata);
76  audio_manager->audio_callback(reinterpret_cast<int16_t*>(stream), len / 2);
77  };
78 
79  // convert device name to valid parameter for sdl call
80  // if the device name is empty, use a nullptr in order to indicate that the
81  // default device should be used
82  const char *c_device_name = device_name.empty() ?
83  nullptr : device_name.c_str();
84  // open audio playback device
85  device_id = SDL_OpenAudioDevice(c_device_name, 0, &desired_spec,
86  &device_spec, 0);
87 
88  // no device could be opened
89  if (device_id == 0) {
90  log::log(MSG(err) << "Error opening audio device: " << SDL_GetError());
91  return;
92  }
93 
94  // initialize playing sounds vectors
95  using sound_vector = std::vector<std::shared_ptr<SoundImpl>>;
96  playing_sounds.insert({category_t::GAME, sound_vector{}});
97  playing_sounds.insert({category_t::INTERFACE, sound_vector{}});
98  playing_sounds.insert({category_t::MUSIC, sound_vector{}});
99  playing_sounds.insert({category_t::TAUNT, sound_vector{}});
100 
101  // create buffer for mixing
102  this->mix_buffer = std::make_unique<int32_t[]>(
103  4 * device_spec.samples * device_spec.channels
104  );
105 
106  log::log(MSG(info) <<
107  "Using audio device: "
108  << (device_name.empty() ? "default" : device_name)
109  << " [freq=" << device_spec.freq
110  << ", format=" << device_spec.format
111  << ", channels=" << static_cast<uint16_t>(device_spec.channels)
112  << ", samples=" << device_spec.samples
113  << "]");
114 
115  SDL_PauseAudioDevice(device_id, 0);
116 
117  this->available = true;
118 }

124 void AudioManager::load_resources(const std::vector<resource_def> &sound_files) {
125  if (not this->available) {

129  for (auto &sound_file : sound_files) {

163 void AudioManager::audio_callback(int16_t *stream, int length) {
164  std::memset(mix_buffer.get(), 0, length*4);
165 
166  // iterate over all categories
167  for (auto &entry : this->playing_sounds) {
168  auto &playing_list = entry.second;
169  // iterate over all sounds in one category
170  for (size_t i = 0; i < playing_list.size(); i++) {
171  auto &sound = playing_list[i];
172  auto sound_finished = sound->mix_audio(mix_buffer.get(), length);
173  // if the sound is finished,
174  // it should be removed from the playing list
175  if (sound_finished) {
176  util::vector_remove_swap_end(playing_list, i);
177  i--;
178  }
179  }
180  }
181 
182  // write the mix buffer to the output stream and adjust volume
183  for (int i = 0; i < length; i++) {
184  auto value = mix_buffer[i]/256;
185  if (value > 32767) {
186  value = 32767;
187  } else if (value < -32768) {
188  value = -32768;
189  }
190  stream[i] = static_cast<int16_t>(value);
191  }
192 }
