40  resources::shader_stage_t::vertex,
41  R"s(
42 #version 330
43 
44 layout(location=0) in vec2 position;
45 layout(location=1) in vec2 uv;
46 uniform mat4 mv;
47 uniform mat4 proj;
48 out vec2 v_uv;
49

53 }
54 )s");

178  /* Make a framebuffer for the first render pass to draw into. The framebuffer consists of a color texture
179  * to be copied onto the back buffer in pass 2, as well as an id texture which will contain the object ids
180  * which we can later read in order to determine which object was clicked. The depth texture is required,

184  auto id_texture = renderer->add_texture(resources::Texture2dInfo(size.first, size.second, resources::pixel_format::r32ui));

188  /* Make an object to update the projection matrix in pass 1 according to changes in the screen size.
189  * Because uniform values are preserved across objects using the same shader in a single render pass,
190  * it is sufficient to set it once at the beginning of the pass. To do this, we use an object with no
191  * geometry, which will set the uniform but not render anything. */

247  window.add_resize_callback([&] (size_t w, size_t h) {
248  /* Calculate a projection matrix for the new screen size. */
249  float aspectRatio = float(w)/float(h);
250  float xScale = 1.0/aspectRatio;
251 
252  Eigen::Matrix4f pmat;

254  0, 1, 0, 0,
255  0, 0, 1, 0,

259 
260  /* Create a new FBO with the right dimensions. */
261  color_texture = renderer->add_texture(resources::Texture2dInfo(w, h, resources::pixel_format::rgba8));
262  id_texture = renderer->add_texture(resources::Texture2dInfo(w, h, resources::pixel_format::r32ui));
263  depth_texture = renderer->add_texture(resources::Texture2dInfo(w, h, resources::pixel_format::depth24));
264  fbo = renderer->create_texture_target( { color_texture.get(), id_texture.get(), depth_texture.get() } );
265 
266  display_shader->update_uniform_input(color_texture_unif.get(), "color_texture", color_texture.get());
267 
268  texture_data_valid = false;
269  pass1.target = fbo.get();
270  } );
271
