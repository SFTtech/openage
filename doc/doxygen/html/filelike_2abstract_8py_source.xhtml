116  def seek_helper(self, offset, whence):
117  """
118  Helper function for use by implementations of seek().
119 
120  Calculates the new cursor position relative to file start
121  from offset, whence and self.tell().
122 
123  If size is given, it works for whence=os.SEEK_END;
124  otherwise, UnsupportedOperation is raised.
125  """
126 
127  if whence == os.SEEK_SET:
128  target = offset
129  elif whence == os.SEEK_CUR:
130  target = offset + self.tell()
131  elif whence == os.SEEK_END:
132  size = self.get_size()
133  if size < 0:
134  raise UnsupportedOperation(
135  "can only seek relative to file start or cursor")
136  target = offset + size
137  else:
138  raise UnsupportedOperation("unsupported seek mode")
139 
140  if target < 0:
141  raise ValueError("can not seek to a negative file position")
