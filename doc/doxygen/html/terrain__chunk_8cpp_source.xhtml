58 TileContent *TerrainChunk::get_data_neigh(coord::tile_delta pos) {
59  // determine the neighbor id by the given position
60  int neighbor_id = this->neighbor_id_by_pos(pos);
61 
62  // if the location is not on the current chunk, the neighbor id is != -1
63  if (neighbor_id != -1) {
64 
65  // get the chunk where the requested neighbor tile lies on.
66  TerrainChunk *neigh_chunk = this->neighbors.neighbor[neighbor_id];
67 
68  // this neighbor does not exist, so the tile does not exist.
69  if (neigh_chunk == nullptr) {
70  return nullptr;
71  }
72 
73  // get position of tile on neighbor
74  size_t pos_on_neighbor = this->tile_position_neigh(pos);
75 
76  return neigh_chunk->get_data(pos_on_neighbor);
77  }
78  // the position lies on the current chunk.
79  else {
80  return this->get_data(pos);
81  }

84 /*
85  * get the chunk neighbor id by a given position not lying on this chunk.
86  *
87  * neighbor ids:
88  *
89  * ne
90  * --
91  * /|
92  * 0
93  * 7 1
94  * 6 @ 2
95  * 5 3
96  * 4
97  * \|
98  * --
99  * se
100  */
101 int TerrainChunk::neighbor_id_by_pos(coord::tile_delta pos) {
102  int neigh_id = -1;
103 
104  if (pos.ne < 0) {
105  if (pos.se < 0) {
106  neigh_id = 6;
107  }
108  else if (pos.se >= (ssize_t)chunk_size) {
109  neigh_id = 4;
110  }
111  else {
112  neigh_id = 5;
113  }
114  }
115  else if (pos.ne >= (ssize_t)chunk_size) {
116  if (pos.se < 0) {
117  neigh_id = 0;
118  }
119  else if (pos.se >= (ssize_t)chunk_size) {
120  neigh_id = 2;
121  }
122  else {
123  neigh_id = 1;
124  }
125  }
126  else {
127  if (pos.se < 0) {
128  neigh_id = 7;
129  }
130  else if (pos.se >= (ssize_t)chunk_size) {
131  neigh_id = 3;
132  }
133  else {
134  neigh_id = -1;
135  }
136  }
137  return neigh_id;
138 }

140 /*
141  * calculates the memory position of a given tile location.
142  *
143  * give this function isometric coordinates, it returns the tile index.
144  *
145  * # is a single terrain tile:
146  *
147  * 3
148  * 2 #
149  * 1 # #
150  * ne= 0 # * #
151  * # # # #
152  * se= 0 # # #
153  * 1 # #
154  * 2 #
155  * 3
156  *
157  * for example, * is at position (2, 1)
158  * the returned index would be 6 (count for each ne row, starting at se=0)
159  */
160 size_t TerrainChunk::tile_position(coord::tile_delta pos) {
161  if (this->neighbor_id_by_pos(pos) != -1) {
162  throw Error(MSG(err) << "Tile "
163  "(" << pos.ne << ", " << pos.se << ") "
164  "has been requested, but is not part of this chunk.");
165  }
166 
167  return pos.se * chunk_size + pos.ne;
168 }
