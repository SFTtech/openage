79 void MoveAbility::invoke(Unit &to_modify, const Command &cmd, bool play_sound) {
80  to_modify.log(MSG(dbg) << "invoke move action");
81  if (play_sound && this->sound) {
82  this->sound->play();
83  }
84 
85  if (cmd.has_position()) {
86  auto target = cmd.position();
87  to_modify.push_action(std::make_unique<MoveAction>(&to_modify, target));
88  }
89  else if (cmd.has_unit()) {
90  auto target = cmd.unit();
91 
92  // distance from the targets edge that is required to stop moving
93  coord::phys_t radius = path::path_grid_size + (to_modify.location->min_axis() / 2L);
94 
95  // add the range of the unit if cmd indicator is set
96  if (cmd.has_flag(command_flag::use_range) && to_modify.has_attribute(attr_type::attack)) {
97  auto &att = to_modify.get_attribute<attr_type::attack>();
98  radius += att.max_range;
99  }
100  to_modify.push_action(std::make_unique<MoveAction>(&to_modify, target->get_ref(), radius));
101  }
102 }

276 bool AttackAbility::can_invoke(Unit &to_modify, const Command &cmd) {
277  if (cmd.has_unit()) {
278  Unit &target = *cmd.unit();
279  bool target_is_resource = has_resource(target);
280  return &to_modify != &target &&
281  to_modify.location &&
282  target.location &&
283  target.location->is_placed() &&
284  to_modify.has_attribute(attr_type::attack) &&
285  has_hitpoints(target) &&
286  (is_enemy(to_modify, target) || target_is_resource) &&
287  (cmd.has_flag(command_flag::attack_res) == target_is_resource);
288  }
289  return false;
