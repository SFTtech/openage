37  def fill(self, data):
38  # split all lines of the input data
39  # \r\n windows windows windows baby
40  lines = data.decode('ascii').split('\r\n')
41 
42  self.header = lines[0]
43  self.version = lines[1]
44 
45  # check for palette header
46  if self.header != "JASC-PAL":
47  raise Exception("No palette header 'JASC-PAL' found, "
48  "instead: %r" % self.header)
49 
50  if self.version != "0100":
51  raise Exception("palette version mispatch, got %s" % self.version)
52 
53  entry_count = int(lines[2])
54 
55  self.palette = []
56 
57  # data entries are line 3 to n
58  for line in lines[3:entry_count + 3]:
59  # one entry looks like "13 37 42",
60  # "red green blue"
61  # => red 13, 37 green and 42 blue.
62  self.palette.append(tuple(int(val) for val in line.split(' ')))
63 
64  if len(self.palette) != len(lines) - 4:
65  raise Exception("read a %d palette entries "
66  "but expected %d." % (
67  len(self.palette), len(lines) - 4))
