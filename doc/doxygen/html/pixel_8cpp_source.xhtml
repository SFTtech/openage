12 phys3_delta camgame_delta::to_phys3(const CoordManager &mgr, phys_t up) const {
13  // apply scaling factor; w/2 for x, h/2 for y
14  phys_t x = phys_t::from_int(this->x) / static_cast<int64_t>(mgr.tile_size.x / 2);
15  phys_t y = phys_t::from_int(this->y) / static_cast<int64_t>(mgr.tile_size.y / 2);
16 
17  // apply transformation matrix to 'scaled',
18  // to get the relative phys3 position
19  //
20  // a camgame position represents a line in the 3D phys space
21  // a camgame delta of 0 might actually correlate to an arbitrarily
22  // large phys delta.
23  // we select one specific point on that line by explicitly specifying the
24  // 'up' value of the result.
25  //
26  // the transformation matrix is:
27  //
28  // (ne) = (+0.5 +0.5 +0.5) ( x)
29  // (se) = (+0.5 -0.5 -0.5) ( y)
30  // (up) = (+0.0 +0.0 +1.0) (up)
31  phys3_delta result;
32  result.ne = (x + y + up)/2L;
33  result.se = (x - y - up)/2L;
34  result.up = ( up);
35 
36  return result;
37 }
