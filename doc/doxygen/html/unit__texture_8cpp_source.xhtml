76 void UnitTexture::draw(const coord::CoordManager &coord, const coord::camgame &draw_pos, coord::phys3_delta &dir, unsigned int frame, unsigned color) const {
77  unsigned int frame_to_use = frame;
78  if (this->use_up_angles) {
79  // up 1 => tilt 0
80  // up -1 => tilt 1
81  // up has a scale 5 times smaller
82  double len = math::hypot3(dir.ne.to_double(), dir.se.to_double(), dir.up.to_double()/5);
83  double up = dir.up.to_double()/(5.0 * len);
84  frame_to_use = (0.5 - (0.5 * up)) * this->frame_count;
85  }
86  else if (this->sound && frame == 0.0) {
87  this->sound->play();
88  }
89 
90  // the index for the current direction
91  unsigned int angle = dir_group(dir, this->angle_count);
92 
93  // mirroring is used to make additional image sets
94  bool mirror = false;
95  if (this->angles_included <= angle) {
96  // this->angles_included <= angle < this->angle_count
97  angle = this->top_frame - angle;
98  mirror = true;
99  }
100 
101  // draw delta list first
102  for (auto &d : this->deltas) {
103  d.first->draw(coord, draw_pos + d.second, dir, frame, color);
104  }
105 
106  if (this->draw_this) {
107  unsigned int to_draw = this->subtexture(this->texture, angle, frame_to_use);
108  this->texture->draw(coord, draw_pos, PLAYERCOLORED, mirror, to_draw, color);
109  }
110 }

112 void UnitTexture::initialise(GameSpec &spec) {
113  this->texture = spec.get_texture(this->id);
114  this->sound = spec.get_sound(this->sound_id);
115  if (not is_valid()) {
116  this->draw_this = false;
117  }
118 
119  // find deltas
120  if (this->use_deltas) for (auto d : this->delta_id) {
121  if (spec.get_graphic_data(d.graphic_id)) {
122  auto ut = std::make_unique<UnitTexture>(spec, d.graphic_id, false);
123  if (ut->is_valid()) {
124  this->deltas.push_back({std::move(ut), coord::camgame_delta{d.offset_x, d.offset_y}});
125  }
126  }
127  }
128 
129  if (this->draw_this) {
130 
131  // the graphic frame count includes deltas
132  unsigned int subtextures = this->texture->get_subtexture_count();
133  if (subtextures >= this->frame_count) {
134 
135  // angles with graphic data
136  this->angles_included = subtextures / this->frame_count;
137  this->angles_mirrored = this->angle_count - this->angles_included;
138  this->safe_frame_count = this->frame_count;
139  }
140  else {
141  this->angles_included = 1;
142  this->angles_mirrored = 0;
143  this->safe_frame_count = subtextures;
144  }
145 
146  // find the top direction for mirroring over
147  this->top_frame = this->angle_count - (1 - (this->angles_included - this->angles_mirrored) / 2);
148  }
149 }

151 unsigned int UnitTexture::subtexture(const Texture *t, unsigned int angle, unsigned int frame) const {
152  unsigned int tex_frames = t->get_subtexture_count();
153  unsigned int count = tex_frames / this->angles_included;
154  unsigned int to_draw = angle * count + (frame % count);
155  if (tex_frames <= to_draw) {
156  log::log(MSG(err) << "Subtexture out of range (" << angle << ", " << frame << ")");
157  return 0;
158  }
159  return to_draw;
160 }

162 unsigned int dir_group(coord::phys3_delta dir, unsigned int angles) {
163  unsigned int first_angle = 5 * angles / 8;
164 
165  // normalise direction vector
166  double len = std::hypot(dir.ne, dir.se);
167  double dir_ne = static_cast<double>(dir.ne) / len;
168  double dir_se = static_cast<double>(dir.se) / len;
169 
170  // formula to find the correct angle
171  return static_cast<unsigned int>(
172  round(angles * atan2(dir_se, dir_ne) * (math::INV_PI / 2))
173  + first_angle
174  ) % angles;
175 }
