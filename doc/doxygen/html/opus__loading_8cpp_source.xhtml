57 opus_file_t open_opus_file(const util::Path &path) {
58  if (not path.is_file()) {
59  throw audio::Error{
60  ERR << "Audio file not found: " << path
61  };
62  }
63 
64  opus_file_t op_file;
65  int op_err = 0;
66 
67  // check if the file can be opened directly
68  auto native_path = path.resolve_native_path();
69  if (native_path.size() > 0) {
70  op_file.handle = {
71  op_open_file(native_path.c_str(), &op_err),
72  opus_deleter
73  };
74  }
75  else {
76  // open the file and move the handle to the heap
77  op_file.file = std::make_unique<util::File>();
78  *op_file.file = path.open_r();
79 
80  op_file.handle = {
81  op_open_callbacks(op_file.file.get(), &opus_access_funcs,
82  nullptr, 0, &op_err),
83  opus_deleter
84  };
85  }
86 
87  if (op_err != 0) {
88 
89  const char *reason;
90 
91  switch (op_err) {
92  case OP_EREAD:
93  reason = "read/seek/tell failed or data has changed"; break;
94  case OP_EFAULT:
95  reason = "opus failed to allocate memory "
96  "or something else bad happened internally"; break;
97  case OP_EIMPL:
98  reason = "Stream used an unsupported feature"; break;
99  case OP_EINVAL:
100  reason = "seek() worked, but tell() did not, "
101  "or initial_bytes != start seek pos"; break;
102  case OP_ENOTFORMAT:
103  reason = "Data didn't contain opus stream"; break;
104  case OP_EBADHEADER:
105  reason = "Header packet was invalid or missing"; break;
106  case OP_EVERSION:
107  reason = "ID header has unrecognized version"; break;
108  case OP_EBADLINK:
109  reason = "Data we already saw before seeking not found"; break;
110  case OP_EBADTIMESTAMP:
111  reason = "Validity check for first/last timestamp failed"; break;
112 
113  default:

115  }
116 
117  throw audio::Error{
118  ERR

120  << path << " = '" << native_path << "': "
121  << reason
122  };
123  }
124  return op_file;
125 }
