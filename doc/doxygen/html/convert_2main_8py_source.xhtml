146 def convert_assets(assets, args, srcdir=None, prev_source_dir_path=None):
147  """
148  Perform asset conversion.
149 
150  Requires original assets and stores them in usable and free formats.
151 
152  assets must be a filesystem-like object pointing at the game's asset dir.
153  srcdir must be None, or point at some source directory.
154

156  are created.

173

175  args.srcdir = AccessSynchronizer(data_dir).root
176  args.targetdir = AccessSynchronizer(targetdir).root

178  def flag(name):
179  """
180  Convenience function for accessing boolean flags in args.
181  Flags default to False if they don't exist.
182  """
183  return getattr(args, name, False)
184 
185  args.flag = flag
186 
187  # import here so codegen.py doesn't depend on it.
188  from .driver import convert
189 
190  converted_count = 0
191  total_count = None
192  for current_item in convert(args):
193  if isinstance(current_item, int):
194  # convert is informing us about the estimated number of remaining
195  # items.
196  total_count = current_item + converted_count

199  # TODO a GUI would be nice here.

379 def source_dir_proposals(call_wine):
380  """Yield a list of directory names where an installation might be found"""
381  if "WINEPREFIX" in os.environ:
382  yield "$WINEPREFIX/" + STANDARD_PATH_IN_32BIT_WINEPREFIX
383  yield "$WINEPREFIX/" + STANDARD_PATH_IN_64BIT_WINEPREFIX
384  yield "$WINEPREFIX/" + STANDARD_PATH_IN_WINEPREFIX_STEAM
385  yield "~/.wine/" + STANDARD_PATH_IN_32BIT_WINEPREFIX
386  yield "~/.wine/" + STANDARD_PATH_IN_64BIT_WINEPREFIX
387  yield "~/.wine/" + STANDARD_PATH_IN_WINEPREFIX_STEAM
388  yield "~/.steam/steam/steamapps/common/Age2HD"
389 
390  if not call_wine:
391  # user wants wine not to be called
392  return
393 
394  try:
395  info("using the wine registry to query an installation location...")
396  # get wine registry key of the age installation
397  with NamedTemporaryFile(mode='rb') as reg_file:
398  if not subprocess.call(('wine', 'regedit', '/E', reg_file.name,
399  REGISTRY_KEY)):
400 
401  reg_raw_data = reg_file.read()
402  try:
403  reg_data = reg_raw_data.decode('utf-16')
404  except UnicodeDecodeError:
405  # this is hopefully enough.
406  # if it isn't, feel free to fight more encoding problems.
407  reg_data = reg_raw_data.decode('utf-8', errors='replace')
408 
409  # strip the REGEDIT4 header, so it becomes a valid INI
410  lines = reg_data.splitlines()
411  del lines[0:2]
412 
413  reg_parser = ConfigParser()
414  reg_parser.read_string(''.join(lines))
415  for suffix in REGISTRY_SUFFIX_AOK, REGISTRY_SUFFIX_TC:
416  reg_key = REGISTRY_KEY + suffix
417  if reg_key in reg_parser:
418  if '"InstallationDirectory"' in reg_parser[reg_key]:
419  yield wine_to_real_path(unescape_winereg(
420  reg_parser[reg_key]['"InstallationDirectory"']))
421  if '"EXE Path"' in reg_parser[reg_key]:
422  yield wine_to_real_path(unescape_winereg(
423  reg_parser[reg_key]['"EXE Path"']))
424 
425  except OSError as error:
426  dbg("wine registry extraction failed: %s", error)

429 def conversion_required(asset_dir, args):
430  """
431  Returns true if an asset conversion is required to run the game.
432 
433  Sets options in args according to what sorts of conversion are required.
434  """
435 
436  version_path = asset_dir / 'converted' / changelog.ASSET_VERSION_FILENAME
437  spec_path = asset_dir / 'converted' / changelog.GAMESPEC_VERSION_FILENAME
438 
439  # determine the version of assets
440  try:
441  with version_path.open() as fileobj:
442  asset_version = fileobj.read().strip()
443 
444  try:
445  asset_version = int(asset_version)
446  except ValueError:
447  info("Converted asset version has improper format; "
448  "expected integer number")
449  asset_version = -1
450 
451  except FileNotFoundError:
452  # assets have not been converted yet
453  info("No converted assets have been found")
454  asset_version = -1
455 
456  # determine the version of the gamespec format
457  try:
458  with spec_path.open() as fileobj:
459  spec_version = fileobj.read().strip()
460 
461  except FileNotFoundError:
462  info("Game specification version file not found.")
463  spec_version = None
464 
465  # TODO: datapack parsing
466  changes = changelog.changes(asset_version, spec_version)
467 
468  if not changes:
469  dbg("Converted assets are up to date")
470  return False
471 
472  if asset_version >= 0 and asset_version != changelog.ASSET_VERSION:
473  info("Found converted assets with version %d, "
474  "but need version %d", asset_version, changelog.ASSET_VERSION)
475 
476  info("Converting %s", ", ".join(sorted(changes)))
477 
478  # try to resolve resolve the output path
479  target_path = asset_dir.resolve_native_path_w()
480  if not target_path:
481  raise OSError("could not resolve a writable asset path "
482  "in {}".format(asset_dir))
483 
484  info("Will save to '%s'", target_path.decode(errors="replace"))
485 
486  for component in changelog.COMPONENTS:
487  if component not in changes:
488  # don't reconvert this component:
489  setattr(args, "no_{}".format(component), True)
490 
491  if "metadata" in changes:
492  args.no_pickle_cache = True
493 
494  return True
