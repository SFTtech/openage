258  std::unique_ptr<PersistentCoreHolderBase> instantiate_core() {
259  T *origin = checked_static_cast<T*>(this);
260 
261  if (origin->holder) {
262  return std::unique_ptr<PersistentCoreHolderBase>();
263  } else {
264  auto core = std::make_unique<typename Unwrap<T>::Type>(origin);
265  return std::make_unique<typename std::remove_reference<decltype(*origin->holder)>::type>(std::move(core));
266  }
267  }

272  void do_adopt_core(PersistentCoreHolderBase *holder, const QString &tag) {
273  assert(holder);
274 
275  T *origin = checked_static_cast<T*>(this);
276 
277  if (origin->holder) {
278  qFatal("Error in QML code: GuiLiveReloader was asked to use same tag '%s' for multiple objects.", qUtf8Printable(tag));
279  } else {
280  if (typeid(decltype(*origin->holder)) != typeid(*holder)) {
281  qFatal(
282  "Error in QML code: GuiLiveReloader was asked "
283  "to restore '%s' into different type '%s' "
284  "using tag '%s'.",
285  qUtf8Printable(name_tidier(typeid(decltype(*origin->holder)).name())),
286  qUtf8Printable(name_tidier(typeid(*holder).name())),
287  qUtf8Printable(tag)
288  );
289  } else {
290  origin->holder = checked_static_cast<decltype(origin->holder)>(holder);
291  origin->holder->adopt_shell(origin);
292  }
293  }
294  }
