33 void EventHandlingQuickWindow::on_input_event(std::atomic<bool> *processed, QEvent *event, bool only_if_grabbed) {
34  if (!only_if_grabbed || this->mouseGrabberItem()) {
35  if (this->focused_item && (event->type() == QEvent::KeyPress || event->type() == QEvent::KeyRelease)) {
36  QCoreApplication::instance()->sendEvent(this->focused_item, event);
37  } else {
38  QCoreApplication::instance()->sendEvent(this, event);
39 
40  auto change_focus = [this] (QQuickItem *item) {
41  if (this->focused_item != item) {
42  if (this->focused_item) {
43  QFocusEvent focus_out{QEvent::FocusOut, Qt::ActiveWindowFocusReason};
44  QCoreApplication::instance()->sendEvent(this->focused_item, &focus_out);
45  }
46 
47  if (item) {
48  QFocusEvent focus_in{QEvent::FocusIn, Qt::ActiveWindowFocusReason};
49  QCoreApplication::instance()->sendEvent(item, &focus_in);
50  }
51  }
52 
53  this->focused_item = item;
54  };
55 
56  // Loose keyboard focus when clicked outside of gui.
57  if (event->type() == QEvent::MouseButtonPress && !event->isAccepted())
58  change_focus(nullptr);
59 
60  // Normally, the QQuickWindow would handle keyboard focus automatically, but it can't because neither QQuickWindow nor
61  // its target QWindow respond to requestActivate(). Which means no focus event propagation when injecting mouse clicks.
62  // So, the workaround is to look specifically for TextFields and give them focus directly.
63  // \todo to remove when the proper focus for the foreign (that obtained from QWindow::fromWinId()) windows is implemented (Qt 5.6).
64  if (this->mouseGrabberItem() && this->mouseGrabberItem()->metaObject()->superClass() && this->mouseGrabberItem()->metaObject()->superClass()->className() == QString("QQuickTextInput") && (event->type() == QEvent::MouseButtonPress))
65  change_focus(this->mouseGrabberItem());
66  }
67  }
68 
69  *processed = true;
70 }

145 GLuint GuiRendererImpl::render() {
146  GuiRenderingCtxActivator activate_render(this->gui_rendering_setup_routines);
147 
148  this->reinit_fbo_if_needed();
149 
150  // QQuickRenderControl::sync() must be called from the render thread while the gui thread is stopped.
151  if (this->need_sync) {
152  if (QCoreApplication::instance()->thread() != QThread::currentThread()) {
153  std::unique_lock<std::mutex> lck{this->gui_guard};
154 
155  if (this->need_sync) {
156  QCoreApplication::instance()->postEvent(this, new QEvent{QEvent::User}, INT_MAX);
157 
158  this->renderer_waiting_on_cond = true;
159  this->gui_locked_cond.wait(lck, [this] {return this->gui_locked;});
160  this->renderer_waiting_on_cond = false;
161 
162  this->render_control.sync();
163 
164  this->need_sync = false;
165  this->gui_locked = false;
166 
167  lck.unlock();
168  this->gui_locked_cond.notify_one();
169  }
170  } else {
171  this->render_control.sync();
172  }
173  }
174 
175  this->render_control.render();
176 
177  this->window->resetOpenGLState();
178 
179  return this->fbo->texture();
180 }

195 bool GuiRendererImpl::make_sure_render_thread_wont_sync() {
196  assert(QThread::currentThread() == QCoreApplication::instance()->thread());
197 
198  if (this->need_sync && QThread::currentThread() != this->render_control.thread()) {
199  std::unique_lock<std::mutex> lck{this->gui_guard};
200

202  this->process_freeze(std::move(lck));
