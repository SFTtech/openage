20 static gl_context_capabilities find_capabilities() {
21  // This is really hacky. We try to create a context starting with
22  // the lowest GL version and retry until one version is not supported and fails.
23  // There is no other way to do this. (https://gamedev.stackexchange.com/a/28457)
24 
25  SDL_Window *test_window = SDL_CreateWindow("test", 0, 0, 2, 2, SDL_WINDOW_OPENGL | SDL_WINDOW_HIDDEN);
26  if (test_window == nullptr) {
27  throw Error(MSG(err) << "Failed creating window for OpenGL context testing. SDL Error: " << SDL_GetError());
28  }
29 
30  // Check each version for availability
31  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
32  SDL_GL_SetAttribute(SDL_GL_ACCELERATED_VISUAL, 1);
33  SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
34  SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, 24);
35 
36  SDL_GLContext test_context;
37  for (size_t i_ver = 0; i_ver < gl_versions.size(); ++i_ver) {
38  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, gl_versions[i_ver].first);
39  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, gl_versions[i_ver].second);
40  test_context = SDL_GL_CreateContext(test_window);
41 
42  if (test_context == nullptr) {
43  if (i_ver == 0) {
44  throw Error(MSG(err) << "OpenGL version "
45  << gl_versions[0].first << "." << gl_versions[0].second
46  << " is not available. It is the minimal required version.");
47  }
48 
49  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, gl_versions[i_ver - 1].first);
50  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, gl_versions[i_ver - 1].second);
51  break;
52  }
53 
54  SDL_GL_DeleteContext(test_context);
55  }
56 
57  test_context = SDL_GL_CreateContext(test_window);
58  if (test_context == nullptr) {
59  throw Error(MSG(err) << "Failed to create OpenGL context which previously succeeded. This should not happen! SDL Error: " << SDL_GetError());
60  }
61  SDL_GL_MakeCurrent(test_window, test_context);
62 
63  gl_context_capabilities caps{};
64 
65  GLint temp;
66  glGetIntegerv(GL_MAX_TEXTURE_SIZE, &temp);
67  caps.max_texture_size = temp;
68  // TODO maybe GL_MAX_TEXTURE_IMAGE_UNITS or maybe GL_MAX_VERTEX_TEXTURE_IMAGE_UNITS
69  // lol opengl
70  glGetIntegerv(GL_MAX_COMBINED_TEXTURE_IMAGE_UNITS, &temp);
71  caps.max_texture_slots = temp;
72  glGetIntegerv(GL_MAX_VERTEX_ATTRIBS, &temp);
73  caps.max_vertex_attributes = temp;
74  glGetIntegerv(GL_MAX_UNIFORM_BUFFER_BINDINGS, &temp);
75  caps.max_uniform_buffer_bindings = temp;
76 
77  glGetIntegerv(GL_MAJOR_VERSION, &caps.major_version);
78  glGetIntegerv(GL_MINOR_VERSION, &caps.minor_version);
79 
80  SDL_GL_DeleteContext(test_context);
81  SDL_DestroyWindow(test_window);
82 
83  return caps;
84 }

86 GlContext::GlContext(SDL_Window *window) {
87  this->capabilities = find_capabilities();
88  auto const &capabilities = this->capabilities;
89 
90  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
91  SDL_GL_SetAttribute(SDL_GL_ACCELERATED_VISUAL, 1);
92  SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
93  SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, 24);
94  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, capabilities.major_version);
95  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, capabilities.minor_version);
96 
97  this->gl_context = SDL_GL_CreateContext(window);
98 
99  if (this->gl_context == nullptr) {
100  throw Error(MSG(err) << "OpenGL context creation failed. SDL error: " << SDL_GetError());
101  }
102 
103  // We still have to verify that our version of libepoxy supports this version of OpenGL.
104  int epoxy_glv = capabilities.major_version * 10 + capabilities.minor_version;
105  if (!epoxy_is_desktop_gl() || epoxy_gl_version() < epoxy_glv) {
106  throw Error(MSG(err) << "The used version of libepoxy does not support OpenGL version "
107  << capabilities.major_version << "." << capabilities.minor_version);
108  }
109 
110  log::log(MSG(info) << "Created OpenGL context version " << capabilities.major_version << "." << capabilities.minor_version);
111 
112  // To quote the standard doc: 'The value gives a rough estimate of the
113  // largest texture that the GL can handle'
114  // -> wat? anyways, we need at least 1024x1024.
115  log::log(MSG(dbg) << "Maximum supported texture size: "
116  << capabilities.max_texture_size);
117  if (capabilities.max_texture_size < 1024) {
118  throw Error(MSG(err) << "Maximum supported texture size is too small: "
119  << capabilities.max_texture_size);
120  }
121 
122  log::log(MSG(dbg) << "Maximum supported texture units: "
123  << capabilities.max_texture_slots);
124  if (capabilities.max_texture_slots < 2) {
125  throw Error(MSG(err) << "Your GPU doesn't have enough texture units: "
126  << capabilities.max_texture_slots);
127  }
128 }

158 void GlContext::check_error() {
159  GLenum error_state = glGetError();
160  if (error_state != GL_NO_ERROR) {
161  const char *msg = [=] {
162  // generate error message
163  switch (error_state) {
164  case GL_INVALID_ENUM:
165  // An unacceptable value is specified for an enumerated argument.
166  // The offending command is ignored
167  // and has no other side effect than to set the error flag.
168  return "GL_INVALID_ENUM";
169  case GL_INVALID_VALUE:
170  // A numeric argument is out of range.
171  // The offending command is ignored
172  // and has no other side effect than to set the error flag.
173  return "GL_INVALID_VALUE";
174  case GL_INVALID_OPERATION:
175  // The specified operation is not allowed in the current state.
176  // The offending command is ignored
177  // and has no other side effect than to set the error flag.
178  return "GL_INVALID_OPERATION";
179  case GL_INVALID_FRAMEBUFFER_OPERATION:
180  // The framebuffer object is not complete. The offending command
181  // is ignored and has no other side effect than to set the error flag.
182  return "GL_INVALID_FRAMEBUFFER_OPERATION";
183  case GL_OUT_OF_MEMORY:
184  // There is not enough memory left to execute the command.
185  // The state of the GL is undefined,
186  // except for the state of the error flags,
187  // after this error is recorded.
188  return "GL_OUT_OF_MEMORY";
189  case GL_STACK_UNDERFLOW:
190  // An attempt has been made to perform an operation that would
191  // cause an internal stack to underflow.
192  return "GL_STACK_UNDERFLOW";
193  case GL_STACK_OVERFLOW:
194  // An attempt has been made to perform an operation that would
195  // cause an internal stack to overflow.
196  return "GL_STACK_OVERFLOW";
197  default:
198  // unknown error state
199  return "unknown error";
200  }
201  }();
202 
203  throw Error(
204  MSG(err) << "An OpenGL error has occured.\n\t"
205  << "(" << error_state << "): " << msg
206  );
207  }
208 }
