52

59  }
60

63  or (p.flags & CHR_BLINKINGFAST and not fastblinking_visible)) {

91 void to_terminal(Buf *buf, util::FD *fd, bool clear) {
92  //move cursor, draw top left corner
93  fd->puts("\x1b[H\u250c");
94  if (clear) {
95  fd->puts("\x1b[J");
96  }
97  //draw top line, including title
98  for (coord::term_t x = 0; x < buf->dims.x; x++) {
99  if (x >= 3 && (x - 3) < (int) buf->title.size()) {
100  fd->putcp(buf->title[x - 3]);
101  } else if (x == 2) {
102  fd->putbyte('[');
103  } else if (x - 3 == (int) buf->title.size()) {
104  fd->putbyte(']');
105  } else {
106  fd->puts("\u2500");
107  }
108  }
109  //draw top right corner
110  fd->puts("\u2510\n");
111  //calculate pos/size of scrollbar
112  //scrollbar_top is the first line that has the scrollbar displayed
113  //scrollbar_bottom is the first line that doesn't have the scrollbar displayed
114  coord::term_t lines_total = buf->scrollback_possible + buf->dims.y;
115  coord::term_t pos_total = buf->scrollback_possible - buf->scrollback_pos;
116  coord::term_t scrollbar_top = (buf->dims.y * pos_total) / lines_total;
117  coord::term_t scrollbar_bottom = (buf->dims.y * (pos_total + buf->dims.y)) / lines_total;
118  if (scrollbar_bottom == scrollbar_top) {
119  if (scrollbar_bottom < buf->dims.y) {
120  scrollbar_bottom++;
121  } else {
122  scrollbar_top--;
123  }
124  }
125 
126  //print lines -scrollback_pos to dims.y - scrollback_pos - 1
127  for (coord::term_t y = 0; y < buf->dims.y; y++) {
128  //draw left line
129  fd->puts("\u2502");
130  //draw chars of this line
131  for (coord::term_t x = 0; x < buf->dims.x; x++) {
132  buf_char p = *(buf->chrdataptr({x, y - buf->scrollback_pos}));
133  if (p.cp < 32) {
134  p.cp = '?';
135  }
136  fd->printf("\x1b[38;5;%dm\x1b[48;5;%dm", p.fgcol, p.bgcol);
137  if (p.flags & CHR_BOLD) {
138  fd->puts("\x1b[1m");
139  }
140  if (p.flags & CHR_BLINKING) {
141  fd->puts("\x1b[5m");
142  }
143  bool cursor_visible_at_current_pos = buf->cursorpos == coord::term{x, y - buf->scrollback_pos};
144  cursor_visible_at_current_pos &= buf->cursor_visible;
145  if ((p.flags & CHR_NEGATIVE) xor cursor_visible_at_current_pos) {
146  //print char negative
147  fd->puts("\x1b[7m");
148  }
149  fd->putcp(p.cp);
150  fd->puts("\x1b[m");
151  }
152  //draw right line
153  if (y >= scrollbar_top and y < scrollbar_bottom) {
154  //draw scrollbar on this part of right line
155  fd->puts("\u2503");
156  } else {
157  fd->puts("\u2502");
158  }
159  fd->putbyte('\n');
160  }
161  //draw bottom left corner
162  fd->puts("\u2514");
163  //draw bottom line
164  for (coord::term_t x = 0; x < buf->dims.x; x++) {
165  fd->puts("\u2500");
166  }
167  //draw bottom right corner
168  fd->puts("\u2518\n");
169 }
