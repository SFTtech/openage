29  VkSurfaceFormatKHR choose_best_surface_format(std::vector<VkSurfaceFormatKHR> const& formats) {
30  // If the implementation doesn't have preferred formats, choose our own
31  if (formats.size() == 1
32  && formats[0].format == VK_FORMAT_UNDEFINED)
33  {
34  return { VK_FORMAT_B8G8R8A8_UNORM, VK_COLOR_SPACE_SRGB_NONLINEAR_KHR };
35  }
36 
37  // Otherwise if one of the preferred formats is good, choose that
38  for (const auto& fmt : formats) {
39  if (fmt.format == VK_FORMAT_B8G8R8_UNORM
40  && fmt.colorSpace == VK_COLOR_SPACE_SRGB_NONLINEAR_KHR)
41  {
42  return fmt;
43  }
44  }
45 
46  // Otherwise select any format
47  return formats[0];
48  }

50  VkPresentModeKHR choose_best_present_mode(std::vector<VkPresentModeKHR> const& modes) {
51  VkPresentModeKHR ret = VK_PRESENT_MODE_MAILBOX_KHR;
52 
53  for (const auto& mode : modes) {
54  if (mode == VK_PRESENT_MODE_MAILBOX_KHR) {
55  return mode;
56  } else if (mode == VK_PRESENT_MODE_FIFO_KHR) {
57  ret = mode;
58  } else if (mode == VK_PRESENT_MODE_FIFO_RELAXED_KHR
59  && ret != VK_PRESENT_MODE_FIFO_KHR)
60  {
61  ret = mode;
62  } else if (mode == VK_PRESENT_MODE_IMMEDIATE_KHR
63  && ret != VK_PRESENT_MODE_FIFO_KHR
64  && ret != VK_PRESENT_MODE_FIFO_RELAXED_KHR)
65  {
66  ret = mode;
67  }
68  }
69 
70  return ret;
71  }

73  VlkDrawableDisplay(VkDevice device, SurfaceSupportDetails details)
74  : device(device) {
75  VkSurfaceFormatKHR format = choose_best_surface_format(details.surface_formats);
76  this->format = format.format;
77 
78  VkPresentModeKHR mode = choose_best_present_mode(details.present_modes);
79 
80  if (details.surface_caps.currentExtent.width != std::numeric_limits<uint32_t>::max()) {
81  this->extent = details.surface_caps.currentExtent;
82  } else {
83  // \todo choose a size from this->size in this case
84  throw Error(MSG(err) << "Window manager does not provide a window size.");
85  }
86 
87  uint32_t img_count = details.surface_caps.minImageCount + 1;
88  if (details.surface_caps.maxImageCount != 0) {
89  img_count = std::max(img_count, details.surface_caps.maxImageCount);
90  }
91 
92  VkSwapchainCreateInfoKHR cr_swap = {};
93  cr_swap.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
94  cr_swap.surface = details.surface;
95  cr_swap.minImageCount = img_count;
96  cr_swap.imageFormat = this->format;
97  cr_swap.imageColorSpace = format.colorSpace;
98  cr_swap.presentMode = mode;
99  // \todo why doesn't validation work?
100  cr_swap.imageExtent = this->extent;
101  cr_swap.imageArrayLayers = 1;
102  // or VK_IMAGE_USAGE_TRANSFER_DST_BIT if not drawing directly (postprocess)
103  cr_swap.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT;
104 
105  if (details.maybe_present_fam) {
106  // We have to share the swapchain between different queues in this case
107  cr_swap.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
108  cr_swap.queueFamilyIndexCount = 2;
109  std::array<uint32_t, 2> q_fams = {{ details.graphics_fam, *details.maybe_present_fam }};
110  cr_swap.pQueueFamilyIndices = q_fams.data();
111  } else {
112  // Otherwise only one queue can access it
113  cr_swap.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
114  }
115 
116  cr_swap.preTransform = details.surface_caps.currentTransform;
117  cr_swap.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR;
118  // Discard pixels which are not visible
119  cr_swap.clipped = VK_TRUE;
120  cr_swap.oldSwapchain = VK_NULL_HANDLE;
121 
122  VK_CALL_CHECKED(vkCreateSwapchainKHR, this->device, &cr_swap, nullptr, &this->swapchain);
123 
124  this->images = vk_do_ritual(vkGetSwapchainImagesKHR, this->device, this->swapchain);
125 
126  // \todo move out?
127  for (auto img : this->images) {
128  VkImageViewCreateInfo cr_view = {};
129  cr_view.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
130  cr_view.image = img;
131  cr_view.viewType = VK_IMAGE_VIEW_TYPE_2D;
132  cr_view.format = this->format;
133  cr_view.components.r = VK_COMPONENT_SWIZZLE_IDENTITY;
134  cr_view.components.g = VK_COMPONENT_SWIZZLE_IDENTITY;
135  cr_view.components.b = VK_COMPONENT_SWIZZLE_IDENTITY;
136  cr_view.components.a = VK_COMPONENT_SWIZZLE_IDENTITY;
137  cr_view.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
138  cr_view.subresourceRange.baseMipLevel = 0;
139  cr_view.subresourceRange.levelCount = 1;
140  cr_view.subresourceRange.baseArrayLayer = 0;
141  cr_view.subresourceRange.layerCount = 1;
142 
143  VkImageView view;
144  VK_CALL_CHECKED(vkCreateImageView, this->device, &cr_view, nullptr, &view);
145 
146  this->image_views.push_back(view);
147  }
148  }
149

151  vkDestroySwapchainKHR(this->device, this->swapchain, nullptr);
152  }
