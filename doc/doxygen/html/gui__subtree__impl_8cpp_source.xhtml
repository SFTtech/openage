132 void GuiSubtreeImpl::component_status_changed(QQmlComponent::Status status) {
133  if (QQmlComponent::Error == status) {
134  qCritical("%s", qUtf8Printable(this->root_component->errorString()));
135  return;
136  }
137 
138  if (QQmlComponent::Ready == status) {
139  assert(!this->root);
140 
141  this->root = qobject_cast<QQuickItem*>(this->root_component->beginCreate(this->engine.rootContext()));
142  assert(this->root);
143 
144  this->init_persistent_items();
145 
146  this->root_component->completeCreate();
147 
148  this->reparent_root();
149  }
150 }
