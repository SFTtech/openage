242 bool Engine::on_resize(coord::viewport_delta new_size) {
243  // \todo move all this to the renderer!
244 
245  log::log(MSG(dbg) << "engine window resize to "
246  << new_size.x << "x" << new_size.y);
247 
248  // update engine window size
249  this->coord.viewport_size = new_size;
250 
251  // update camgame window position, set it to center.
252  this->coord.camgame_viewport = coord::viewport{0, 0} + (this->coord.viewport_size / 2);
253 
254  // update camhud window position, set to lower left corner
255  this->coord.camhud_viewport = {0, coord::pixel_t{this->coord.viewport_size.y}};
256 
257  // reset previous projection matrix
258  glMatrixMode(GL_PROJECTION);
259  glLoadIdentity();
260 
261  // update OpenGL viewport: the rendering area
262  glViewport(0, 0, this->coord.viewport_size.x, this->coord.viewport_size.y);
263 
264  // set orthographic projection: left, right, bottom, top, near_val, far_val
265  glOrtho(0, this->coord.viewport_size.x, 0, this->coord.viewport_size.y, 9001, -1);
266 
267  // reset the modelview matrix
268  glMatrixMode(GL_MODELVIEW);
269  glLoadIdentity();
270 
271  return true;
272 }

315 void Engine::loop() {
316  SDL_Event event;
317  util::Timer cap_timer;
318 
319  while (this->running) {
320  this->profiler.start_frame_measure();
321  this->fps_counter.frame();
322  cap_timer.reset(false);
323 
324  this->job_manager.execute_callbacks();
325 
326  this->profiler.start_measure("events", {1.0, 0.0, 0.0});
327  // top level input handling
328  while (SDL_PollEvent(&event)) {
329  switch (event.type) {
330 
331  case SDL_QUIT:
332  this->stop();
333  break;
334 
335  case SDL_WINDOWEVENT: {
336  if (event.window.event == SDL_WINDOWEVENT_RESIZED) {
337  coord::viewport_delta new_size{event.window.data1, event.window.data2};
338 
339  // call additional handlers for the resize event
340  for (auto &handler : on_resize_handler) {
341  if (!handler->on_resize(new_size)) {
342  break;
343  }
344  }
345  }
346  break;
347  }
348 
349  default:
350  for (auto &action : this->on_input_event) {
351  if (false == action->on_input(&event)) {
352  break;
353  }
354  }
355  } // switch event
356  }
357 
358  // here, call to Qt and process all the gui events.
359  this->gui->process_events();
360 
361  if (this->game) {
362  // read camera movement input keys and/or cursor location,
363  // and move camera accordingly.
364 
365  // camera movement speed, in pixels per millisecond
366  // one pixel per millisecond equals 14.3 tiles/second
367  float mov_x = 0.0, mov_y = 0.0, cam_movement_speed_keyboard = 0.5;
368  Uint32 win_flags = SDL_GetWindowFlags(this->window);
369  bool inp_focus = win_flags & SDL_WINDOW_INPUT_FOCUS;
370 
371  input::InputManager &input = this->get_input_manager();
372  using Edge = input::InputManager::Edge;
373 
374  if (inp_focus and (input.is_down(SDLK_LEFT) or
375  input.is_mouse_at_edge(Edge::LEFT, coord.viewport_size.x))) {
376  mov_x = -cam_movement_speed_keyboard;
377  }
378  if (inp_focus and (input.is_down(SDLK_RIGHT) or
379  input.is_mouse_at_edge(Edge::RIGHT, coord.viewport_size.x))) {
380  mov_x = cam_movement_speed_keyboard;
381  }
382  if (inp_focus and (input.is_down(SDLK_DOWN) or
383  input.is_mouse_at_edge(Edge::DOWN, coord.viewport_size.y))) {
384  mov_y = cam_movement_speed_keyboard;
385  }
386  if (inp_focus and (input.is_down(SDLK_UP) or
387  input.is_mouse_at_edge(Edge::UP, coord.viewport_size.y))) {
388  mov_y = -cam_movement_speed_keyboard;
389  }
390 
391  // perform camera movement
392  this->move_phys_camera(
393  mov_x, mov_y,
394  static_cast<float>(this->lastframe_duration_nsec()) / 1e6);
395 
396  // update the currently running game
397  this->game->update(this->lastframe_duration_nsec());
398  }
399  this->profiler.end_measure("events");
400 
401  // call engine tick callback methods
402  for (auto &action : this->on_engine_tick) {
403  if (false == action->on_tick()) {
404  break;
405  }
406  }
407 
408  this->profiler.start_measure("rendering", {0.0, 1.0, 0.0});
409 
410  // clear the framebuffer to black
411  // in the future, we might disable it for lazy drawing
412  glClearColor(0.0, 0.0, 0.0, 0.0);
413  glClear(GL_COLOR_BUFFER_BIT | GL_STENCIL_BUFFER_BIT);
414 
415  // invoke all game drawing handlers
416  for (auto &action : this->on_drawgame) {
417  if (false == action->on_draw()) {
418  break;
419  }
420  }
421 
422  util::gl_check_error();
423 
424  // draw the fps overlay
425  if (this->drawing_debug_overlay.value) {
426  this->draw_debug_overlay();
427  }
428 
429  // invoke all hud drawing callback methods
430  if (this->drawing_huds.value) {
431  for (auto &action : this->on_drawhud) {
432  if (false == action->on_drawhud()) {
433  break;
434  }
435  }
436  }
437 
438  this->text_renderer->render();
439 
440  util::gl_check_error();
441 
442  this->profiler.end_measure("rendering");
443 
444  this->profiler.start_measure("idle", {0.0, 0.0, 1.0});
445 
446  // the rendering is done
447  // swap the drawing buffers to actually show the frame
448  SDL_GL_SwapWindow(window);
449 
450  if (this->ns_per_frame != 0) {
451  uint64_t ns_for_current_frame = cap_timer.getval();
452  if (ns_for_current_frame < this->ns_per_frame) {
453  SDL_Delay((this->ns_per_frame - ns_for_current_frame) / 1e6);
454  }
455  }
456 
457  this->profiler.end_measure("idle");
458 
459  this->profiler.end_frame_measure();
460  }
