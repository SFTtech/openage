66 QObject* GuiImageProviderLink::provider(QQmlEngine *engine, const char *id) {
67  qtsdl::QmlEngineWithSingletonItemsInfo *engine_with_singleton_items_info = qtsdl::checked_static_cast<qtsdl::QmlEngineWithSingletonItemsInfo*>(engine);
68  auto image_providers = engine_with_singleton_items_info->get_image_providers();
69 
70  auto found_it = std::find_if(std::begin(image_providers), std::end(image_providers), [id] (qtsdl::GuiImageProviderImpl *image_provider) {
71  return image_provider->get_id() == id;
72  });
73 
74  ENSURE(found_it != std::end(image_providers), "The image provider '" << id << "' wasn't created or wasn't passed to the QML engine creation function.");
75 
76  // owned by the QML engine
77  return new GuiImageProviderLink{nullptr, *qtsdl::checked_static_cast<GuiGameSpecImageProviderImpl*>(*found_it)};
78 }
