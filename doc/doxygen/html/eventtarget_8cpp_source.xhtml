14 void EventTarget::changes(const curve::time_t &time) {
15  // This target has some change, so we have to notify all dependents
16  // that subscribed on this target.
17 
18  log::log(DBG << "Target: change happens on " << this->id() << " at t=" << time);
19  if (this->parent_notifier != nullptr) {
20  this->parent_notifier(time);
21  }
22 
23  // This is a maybe-erase construct so obsolete dependents are cleaned up.
24  for (auto it = this->dependents.begin(); it != this->dependents.end(); ) {
25  auto dependent = it->lock();
26  if (dependent) {
27  switch (dependent->get_eventclass()->type) {
28  case EventClass::trigger_type::DEPENDENCY_IMMEDIATELY:
29  case EventClass::trigger_type::DEPENDENCY:
30  // Enqueue a change so that change events will be retriggered
31  // (that depend on this target)
32  // FIXME: Traverse the changed value
33  this->loop->create_change(dependent, time, 0);
34  ++it;
35  break;
36 
37  case EventClass::trigger_type::ONCE:
38  // If the dependent is a ONCE-event
39  // Has it been triggered? If so - forget it.
40  if (dependent->last_triggered > 0) {
41  it = this->dependents.erase(it);
42  } else {
43  // FIXME: Traverse the changed value
44  this->loop->create_change(dependent, time, 0);
45  ++it;
46  }
47  break;
48 
49  case EventClass::trigger_type::TRIGGER:
50  case EventClass::trigger_type::REPEAT:
51  // Ignore announced changes for execute/keyframe events
52  ++it;
53  break;
54  }
55  }
56  else {
57  // The dependent is no more, so we can safely forget him
58  it = this->dependents.erase(it);
59  }
60  }
61 }

64 void EventTarget::trigger(const curve::time_t &last_valid_time) {
65  // notify all dependent events that are triggered `on_keyframe`
66  // that the this target changed.
67  // the only events that is "notified" by are TRIGGER.
68 
69  for (auto it = this->dependents.begin(); it != this->dependents.end(); ) {
70  auto dependent = it->lock();
71  if (dependent) {
72  if (dependent->get_eventclass()->type == EventClass::trigger_type::TRIGGER) {
73  log::log(DBG << "Target: trigger creates a change for "
74  << dependent->get_eventclass()->id()
75  << " at t=" << last_valid_time);
76 
77  // FIXME: Traverse the changed value
78  loop->create_change(dependent, last_valid_time, 0);
79  }
80 
81  ++it;
82  }
83  else {
84  it = this->dependents.erase(it);
85  }
86  }
