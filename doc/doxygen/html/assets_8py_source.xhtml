18 def get_asset_path(custom_asset_dir=None):
19  """
20  Returns a Path object for the game assets.
21 
22  `custom_asset_dir` can a custom asset directory, which is mounted at the
23  top of the union filesystem (i.e. has highest priority).
24 
25  This function is used by the both the conversion process
26  and the game startup. The conversion uses it for its output,
27  the game as its data source(s).
28  """
29 
30  # if we're in devmode, use only the in-repo asset folder
31  if not custom_asset_dir and config.DEVMODE:
32  return Directory(os.path.join(config.BUILD_SRC_DIR, "assets")).root
33 
34  # else, mount the possible locations in an union:
35  # overlay the global dir and the user dir.
36  result = Union().root
37 
38  # the cmake-determined folder for storing assets
39  global_data = Path(config.GLOBAL_ASSET_DIR)
40  if global_data.is_dir():
41  result.mount(WriteBlocker(Directory(global_data).root).root)
42 
43  # user-data directory as provided by environment variables
44  # and platform standards
45  # we always create this!
46  home_data = default_dirs.get_dir("data_home") / "openage"
47  result.mount(
48  Directory(
49  home_data,
50  create_if_missing=True
51  ).root / "assets"
52  )
53 
54  # the program argument overrides it all
55  if custom_asset_dir:
56  result.mount(Directory(custom_asset_dir).root)
57 
58  return result
