16 std::shared_ptr<Event> EventQueue::create_event(const std::shared_ptr<EventTarget> &trgt,
17  const std::shared_ptr<EventClass> &cls,
18  const std::shared_ptr<State> &state,
19  const curve::time_t &reference_time,
20  const EventClass::param_map &params) {
21 
22  auto event = std::make_shared<Event>(trgt, cls, params);
23 
24  cls->setup_event(event, state);
25 
26  switch(cls->type) {
27  case EventClass::trigger_type::DEPENDENCY:
28  case EventClass::trigger_type::REPEAT:
29  case EventClass::trigger_type::ONCE:
30  event->set_time(event->get_eventclass()
31  ->predict_invoke_time(trgt, state, reference_time));
32 
33  if (event->get_time() == std::numeric_limits<curve::time_t>::min()) {
34  log::log(DBG << "Queue: ignoring insertion of event "
35  << event->get_eventclass()->id() <<
36  " because no execution was scheduled.");
37 
38  return {};
39  }
40  break;
41 
42  case EventClass::trigger_type::DEPENDENCY_IMMEDIATELY:
43  case EventClass::trigger_type::TRIGGER:
44  event->set_time(reference_time);
45  break;
46  }
47 
48  log::log(DBG << "Queue: inserting event " << event->get_eventclass()->id() <<
49  " into queue to be executed at t=" << event->get_time());
50 
51  // store the event
52  // or enqueue it for execution
53  switch(event->get_eventclass()->type) {
54  case EventClass::trigger_type::DEPENDENCY:
55  this->dependency_events.insert(event);
56  break;
57 
58  case EventClass::trigger_type::DEPENDENCY_IMMEDIATELY:
59  this->dependency_immediately_events.insert(event);
60  break;
61 
62  case EventClass::trigger_type::TRIGGER:
63  this->trigger_events.insert(event);
64  break;
65 
66  case EventClass::trigger_type::REPEAT:
67  case EventClass::trigger_type::ONCE:
68  default:
69  this->event_queue.push(event);
70  }
71 
72  return event;
73 }

82 void EventQueue::add_change(const std::shared_ptr<Event> &event,
83  const curve::time_t &changed_at) {
84 
85  auto it = this->changes->find(OnChangeElement(event, changed_at));
86 
87  // Has the event been triggered in this round?
88  if (event->last_triggered < changed_at) {
89  // Is the change already in the queue?
90  if (it != changes->end()) {
91  // Is the new change dated _before_ the old one?
92  if (it->time > changed_at) {
93  log::log(DBG << "Queue: adjusting time in change queue");
94 
95  // Save the element
96  OnChangeElement e = *it;
97  e.time = changed_at;
98 
99  // delete it from the container and readd it
100  it = this->changes->erase(it);
101  it = this->changes->insert(it, e);
102  }
103  else {
104  // this change is to be ignored
105  log::log(DBG << "Queue: skipping change for " << event->get_eventclass()->id()
106  << " at " << changed_at
107  << " because there was already a better one at t=" << it->time);
108  }
109  }
110  else {
111  // the change was not in the to be changed list
112  this->changes->emplace(event, changed_at);
113  log::log(DBG << "Queue: inserting change for " << event->get_eventclass()->id()
114  << " to be applied at t=" << changed_at);
115  }
116  }
117  else {
118  // the event has been triggered in this round already, so skip it this time
119  this->future_changes->emplace(event, changed_at);
120  log::log(DBG << "Queue: putting change for "
121  << event->get_eventclass()->id() << " into the future, "
122  << "applied at t=" << changed_at);
123  }
124 
125  event->last_triggered = changed_at;
126 }

157 std::shared_ptr<Event> EventQueue::take_event(const curve::time_t &max_time) {
158  if (this->event_queue.size() == 0) {
159  return nullptr;
160  }
161 
162  std::shared_ptr<Event> event = this->event_queue.top();
163 
164  // check if this event should be processed
165  // we take any event that happens <= max_time
166  if (event->get_time() <= max_time) {
167  // remove the event from the queue
168  this->event_queue.pop();
169 
170  return event;
171  }
172  else {
173  return nullptr;
174  }

201 size_t EventQueue::OnChangeElement::Equal::operator()(const OnChangeElement& left,
202  const OnChangeElement& right) const {
203  auto left_evnt = left.evnt.lock();
204  auto right_evnt = right.evnt.lock();
205  if (left_evnt && right_evnt) {
206  if (left_evnt->get_eventclass()->id() == right_evnt->get_eventclass()->id()) {
207  return true;

209  } else {
210  return false;
211  }

214  auto right_trgt = right_evnt->get_target().lock();
215
