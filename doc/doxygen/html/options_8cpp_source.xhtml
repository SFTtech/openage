245 std::vector<std::string> OptionNode::list_options(bool recurse, std::string indent) {
246  std::vector<std::string> result;
247  result.push_back(indent + "node " + this->name + " {");
248  std::string inner_indent = indent + "\t";
249  for (auto &v : varmap) {
250  result.push_back(inner_indent + "var " + v.first + " = " + v.second.str_value());
251  }
252  for (auto &f : actions) {
253  result.push_back(inner_indent + "func " + f.first + "()");
254  }
255  for (auto &p : children) {
256  for (auto &line : p.second->list_options(recurse, inner_indent)) {
257  result.push_back(line);
258  }
259  }
260  result.push_back(indent + "}");
261  return result;
262 }
