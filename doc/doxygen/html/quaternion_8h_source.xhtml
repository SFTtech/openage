65  template <size_t N=4>
66  Quaternion(const Matrix<N, N, T> &mat) {
67 
68  static_assert(N == 3 or N == 4, "only 3 and 4 dimensional matrices can be converted to a quaternion!");
69 
70  T trace = mat.trace();
71  T trace_cmp = trace;
72 
73  if (N == 4) {
74  trace_cmp -= mat[3][3];
75  } else {
76  trace += 1.0;
77  }
78 
79  if (trace_cmp > 0) {
80  T trace_root = std::sqrt(trace); // = 2w
81  this->w = trace_root * 0.5; // = w
82  trace_root = 0.5 / trace_root; // = 1/4w
83 
84  this->x = (mat[2][1] - mat[1][2]) * trace_root;
85  this->y = (mat[0][2] - mat[2][0]) * trace_root;
86  this->z = (mat[1][0] - mat[0][1]) * trace_root;
87  }
88  else {
89  // determine highest diagonal component:
90  int n0 = 0;
91 
92  if (mat[1][1] > mat[0][0]) {
93  n0 = 1;
94  }
95 
96  if (mat[2][2] > mat[n0][n0]) {
97  n0 = 2;
98  }
99 
100  // following indices to 0, 1, 2
101  int next_index[] = {1, 2, 0};
102 
103  // indexable pointers
104  T *ptrs[] = {&this->x, &this->y, &this->z};
105 
106  int n1 = next_index[n0];
107  int n2 = next_index[n1];
108 
109  // this avoids float fuckups:
110  T trace_ordered = (mat[n0][n0] - (mat[n1][n1] + mat[n2][n2]));
111 
112  if (N == 4) {
113  trace_ordered += mat[3][3];
114  } else {
115  trace_ordered += 1.0;
116  }
117 
118  T trace_root = std::sqrt(trace_ordered);
119 
120  *ptrs[n0] = trace_root * 0.5; // = w
121  trace_root = 0.5 / trace_root; // = 1/4w
122 
123  *ptrs[n1] = (mat[n0][n1] + mat[n1][n0]) * trace_root;
124  *ptrs[n2] = (mat[n2][n0] + mat[n0][n2]) * trace_root;
125 
126  this->w = (mat[n2][n1] - mat[n1][n2]) * trace_root;
127  }
128 
129  if (N == 4) {
130  // normalize the quaternion by the matrix[3][3] entry
131  (*this) *= 1.0 / std::sqrt(mat[3][3]);
132  }
133  }
