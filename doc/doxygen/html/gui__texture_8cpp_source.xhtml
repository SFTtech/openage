43 GLuint create_compatible_texture(GLuint texture_id, GLsizei w, GLsizei h) {
44  glBindTexture(GL_TEXTURE_2D, texture_id);
45 
46  GLint min_filter;
47  GLint mag_filter;
48  GLint iformat;
49 
50  glGetTexParameteriv(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, &min_filter);
51  glGetTexParameteriv(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, &mag_filter);
52  glGetTexLevelParameteriv(GL_TEXTURE_2D, 0, GL_TEXTURE_INTERNAL_FORMAT, &iformat);
53 
54  GLuint new_texture_id;
55  glGenTextures(1, &new_texture_id);
56  glBindTexture(GL_TEXTURE_2D, new_texture_id);
57 
58  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, min_filter);
59  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, mag_filter);
60 
61  glTexImage2D(GL_TEXTURE_2D, 0, iformat, w, h, 0, GL_RGBA, GL_UNSIGNED_BYTE, nullptr);
62 
63  glBindTexture(GL_TEXTURE_2D, 0);
64 
65  return new_texture_id;

69 QSGTexture* GuiTexture::removedFromAtlas() const {
70  if (this->isAtlasTexture()) {
71  if (!this->standalone) {
72  auto tex = this->texture_handle.texture;
73  auto sub = tex->get_subtexture(this->texture_handle.subid);
74 
75  GLuint sub_texture_id = create_compatible_texture(tex->get_texture_id(), sub->w, sub->h);
76 
77  std::array<GLuint, 2> fbo;
78  glGenFramebuffers(fbo.size(), &fbo.front());
79 
80  glBindFramebuffer(GL_READ_FRAMEBUFFER, fbo[0]);
81  glFramebufferTexture2D(GL_READ_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, this->textureId(), 0);
82  glReadBuffer(GL_COLOR_ATTACHMENT0);
83 
84  glBindFramebuffer(GL_DRAW_FRAMEBUFFER, fbo[1]);
85  glFramebufferTexture2D(GL_DRAW_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, sub_texture_id, 0);
86  glDrawBuffer(GL_COLOR_ATTACHMENT0);
87 
88  glBlitFramebuffer(sub->x, sub->y, sub->x + sub->w, sub->y + sub->h, 0, 0, sub->w, sub->h, GL_COLOR_BUFFER_BIT, GL_NEAREST);
89 
90  glBindFramebuffer(GL_FRAMEBUFFER, 0);
91 
92  glDeleteFramebuffers(fbo.size(), &fbo.front());
93 
94  this->standalone = make_standalone_subtexture(sub_texture_id, QSize(sub->w, sub->h));
95  }
96 
97  return this->standalone.get();
98  }
99 
100  return nullptr;
101 }
