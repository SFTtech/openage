60 camgame_delta phys3_delta::to_camgame(const CoordManager &mgr) const {
61  // apply transformation matrix to phys3_delta, to get 'scaled':
62  // (ne)
63  // (x) = (+1 +1 +0) (se)
64  // (y) = (+1 -1 +1) (up)
65  phys_t x = phys_t{this->ne} * (+1) + this->se * (+1) + this->up * (+0);
66  phys_t y = phys_t{this->ne} * (+1) + this->se * (-1) + this->up * (+1);
67 
68  // add scaling (w/2 for x, h/2 for y)
69  return camgame_delta{
70  static_cast<camgame_delta::elem_t>((x * (mgr.tile_size.x / 2)).to_int()),
71  static_cast<camgame_delta::elem_t>((y * (mgr.tile_size.y / 2)).to_int())
72  };
73 }
