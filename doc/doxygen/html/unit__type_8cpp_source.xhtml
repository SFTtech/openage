59 TerrainObject *UnitType::place_beside(Unit *u, TerrainObject const *other) const {
60  if (!u || !other) {
61  return nullptr;
62  }
63 
64  // find the range of possible tiles
65  tile_range outline{other->pos.start - coord::tile_delta{1, 1},
66  other->pos.end + coord::tile_delta{1, 1},
67  other->pos.draw};
68 
69  // find a free position adjacent to the object
70  auto terrain = other->get_terrain();
71  for (coord::tile temp_pos : tile_list(outline)) {
72  TerrainChunk *chunk = terrain->get_chunk(temp_pos);
73 
74  if (chunk == nullptr) {
75  continue;
76  }
77 
78  auto placed = this->place(u, terrain, temp_pos.to_phys3(*terrain));
79  if (placed) {
80  return placed;
81  }
82  }
83  return nullptr;
84 }

138 TerrainObject *NyanType::place(Unit *unit, std::shared_ptr<Terrain> terrain, coord::phys3 init_pos) const {
139  // the parsed nyan data gives the rules for terrain placement
140  // which includes valid terrains, base radius and shape
141 
142  unit->make_location<SquareObject>(coord::tile_delta{1, 1});
143 
144  // allow unit to go anywhere
145  unit->location->passable = [](const coord::phys3 &) {
146  return true;
147  };
148 
149  // try to place the obj, it knows best whether it will fit.
150  if (unit->location->place(terrain, init_pos, object_state::placed)) {
151  return unit->location.get();
152  }
153 
154  // placing at the given position failed
