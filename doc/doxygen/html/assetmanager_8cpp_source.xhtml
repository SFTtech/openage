133 void AssetManager::check_updates() {
134 #if WITH_INOTIFY
135  // buffer for at least 4 inotify events
136  char buf[4 * (sizeof(struct inotify_event) + NAME_MAX + 1)];
137  ssize_t len;
138 
139  while (true) {
140  // fetch all events, the kernel won't write "half" structs.
141  len = read(this->inotify_fd, buf, sizeof(buf));
142 
143  if (len == -1) {
144  if (errno == EAGAIN) {
145  // no events, nothing to do.
146  break;
147  } else {
148  // something went wrong
149  log::log(WARN << "Failed to read inotify events!");
150  break;
151  }
152  }
153 
154  // process fetched events,
155  // the kernel guarantees complete events in the buffer.
156  char *ptr = buf;
157  while (ptr < buf + len) {
158  struct inotify_event *event = (struct inotify_event *)ptr;
159 
160  if (event->mask & IN_CLOSE_WRITE) {
161  // TODO: this should invoke callback functions
162  this->watch_fds[event->wd]->reload();
163  }
164 
165  // move the buffer ptr to the next event.
166  ptr += sizeof(struct inotify_event) + event->len;
167  }
168  }
169 #endif
170 }
