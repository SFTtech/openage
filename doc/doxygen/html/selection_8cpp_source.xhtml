136 void UnitSelection::add_unit(const Player &player, Unit *u, bool append) {
137  // Only select resources and units with hitpoints > 0
138  if (u->has_attribute(attr_type::resource) ||
139  (u->has_attribute(attr_type::damaged) && u->get_attribute<attr_type::damaged>().hp > 0)) {
140 
141  selection_type_t unit_type = get_unit_selection_type(player, u);
142  int unit_type_i = static_cast<int>(unit_type);
143  int selection_type_i = static_cast<int>(this->selection_type);
144 
145  if (unit_type_i > selection_type_i) {
146  // Don't select this unit as it has too low priority
147  return;
148  }
149 
150  if (unit_type_i < selection_type_i) {
151  // Upgrade selection to a higher priority selection
152  this->clear();
153  this->selection_type = unit_type;
154  }
155 
156  // Can't select multiple enemies at once
157  if (not (unit_type == selection_type_t::own_units ||
158  (unit_type == selection_type_t::own_buildings && append))) {
159 
160  this->clear();
161  this->selection_type = unit_type; // Clear resets selection_type
162  }
163 
164  // Finally, add the unit to the selection
165  u->selected = true;
166  this->units[u->id] = u->get_ref();
167  }
168 }

246 void UnitSelection::select_space(const Player &player, Terrain *terrain,
247  coord::camgame point0, coord::camgame point1, bool append) {
248  if (!append) {
249  this->clear();
250  }
251 
252  coord::camgame min{std::min(point0.x, point1.x), std::min(point0.y, point1.y)};
253  coord::camgame max{std::max(point0.x, point1.x), std::max(point0.y, point1.y)};
254 
255  // look at each tile in the range and find all units
256  for (coord::tile check_pos : tiles_in_range(point0, point1, this->engine->coord)) {
257  TileContent *tc = terrain->get_data(check_pos);
258  if (tc) {
259  // find objects within selection box
260  for (auto unit_location : tc->obj) {
261  coord::camgame pos = unit_location->pos.draw.to_camgame(this->engine->coord);
262 
263  if ((pos.x > min.x and pos.x < max.x) and
264  (pos.y > min.y and pos.y < max.y)) {
265  this->add_unit(player, &unit_location->unit, append);
266  }
267  }
268  }
269  }

302  auto &build_attr = u->get_attribute<attr_type::building>();
303  lines.push_back("built: "+std::to_string(build_attr.completed));
304  }

306  auto &garrison_attr = u->get_attribute<attr_type::garrison>();

309

315  engine->render_text({0, vpos}, this->font_size, "%s", s.c_str());

331 std::vector<coord::tile> tiles_in_range(coord::camgame p1, coord::camgame p2, const coord::CoordManager &coord) {
332  // the remaining corners
333  coord::camgame p3 = coord::camgame{p1.x, p2.y};
334  coord::camgame p4 = coord::camgame{p2.x, p1.y};
335  coord::camgame pts[4] {p1, p2, p3, p4};
336 
337  // find the range of tiles covered
338  coord::tile t1 = pts[0].to_tile(coord);
339  coord::tile min = t1;
340  coord::tile max = t1;
341 
342  for (unsigned i = 1; i < 4; ++i) {
343  coord::tile t = pts[i].to_tile(coord);
344  min.ne = std::min(min.ne, t.ne);
345  min.se = std::min(min.se, t.se);
346  max.ne = std::max(max.ne, t.ne);
347  max.se = std::max(max.se, t.se);
348  }
349 
350  // find all units in the boxed region
351  std::vector<coord::tile> tiles;
352  coord::tile check_pos = min;
353  while (check_pos.ne <= max.ne) {
354  while (check_pos.se <= max.se) {
355  tiles.push_back(check_pos);
356  check_pos.se += 1;
357  }
358  check_pos.se = min.se;
359  check_pos.ne += 1;
360  }
361  return tiles;
362 }
