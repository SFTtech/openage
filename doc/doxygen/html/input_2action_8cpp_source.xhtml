117 bool ActionManager::create(const std::string& type) {
118  if (this->actions.find(type) != this->actions.end()) {
119  // that action is already in the list. fail.
120  // TODO: throw an exception instead?
121  // for now, just print a warning log message
122  log::log(WARN << "can not create action "
123  << util::repr(type) << ": already exists");
124  return false;
125  }
126 
127  action_t action_id = this->next_action_id++;
128 
129  this->reverse_map[action_id] = type;
130  this->actions[type] = action_id;
131 
132  // and the corresponding cvar, which modifies the action bindings
133  // TODO: this has nothing to do with the actionmanager!
134  // remove the cvarmanager-access here!
135  this->cvar_manager->create(type, std::make_pair(
136  // the cvar's getter
137  [type, this]() {
138  return this->input_manager->get_bind(type);
139  },
140  // the cvar's setter
141  [type, this](const std::string &value) {
142  this->input_manager->set_bind(value.c_str(), type);
143  }
144  ));
145 
146  return true;
147 }
