71 bool TerrainObject::place(object_state init_state) {
72  if (this->state == object_state::removed) {
73  throw Error(MSG(err) << "Building cannot change state with no position");
74  }
75 
76  // remove any other floating objects
77  // which intersect with the new placement
78  // if non-floating objects are on the foundation
79  // then this placement will fail
80  for (coord::tile temp_pos : tile_list(this->pos)) {
81  std::vector<TerrainObject *> to_remove;
82  TerrainChunk *chunk = this->get_terrain()->get_chunk(temp_pos);
83 
84  if (chunk == nullptr) {
85  continue;
86  }
87 
88  for (auto obj : chunk->get_data(temp_pos)->obj) {
89 
90  // ignore self and annexes of self
91  if (obj != this &&
92  obj->get_parent() != this) {
93 
94  if (obj->is_floating()) {
95 
96  // floating objects get removed
97  to_remove.push_back(obj);
98  }
99  else if (obj->check_collisions()) {
100 
101  // solid objects obstruct placement
102  return false;
103  }
104  }
105  }
106 
107  // all obstructing objects get deleted
108  for (auto remove_obj : to_remove) {
109  remove_obj->unit.location = nullptr;
110  }
111  }
112 
113  // set new state
114  this->state = init_state;
115  return true;
116 }

155 void TerrainObject::remove() {
156  // remove all children first
157  for (auto &c : this->children) {
158  c->remove();
159  }
160  this->children.clear();
161 
162  if (this->occupied_chunk_count == 0 ||
163  this->state == object_state::removed) {
164  return;
165  }
166 
167  for (coord::tile temp_pos : tile_list(this->pos)) {
168  TerrainChunk *chunk = this->get_terrain()->get_chunk(temp_pos);
169 
170  if (chunk == nullptr) {
171  continue;
172  }
173 
174  auto &v = chunk->get_data(temp_pos.get_pos_on_chunk())->obj;
175  auto position_it = std::remove_if(
176  std::begin(v),
177  std::end(v),
178  [this](TerrainObject *obj) {
179  return this == obj;
180  });
181  v.erase(position_it, std::end(v));
182  }
183 
184  this->occupied_chunk_count = 0;
185  this->state = object_state::removed;
186 }

188 void TerrainObject::set_ground(int id, int additional) {
189  if (not this->is_placed()) {
190  throw Error(MSG(err) << "Setting ground for object that is not placed yet.");
191  }
192 
193  coord::tile temp_pos = this->pos.start;
194  temp_pos.ne -= additional;
195  temp_pos.se -= additional;
196  while (temp_pos.ne < this->pos.end.ne + additional) {
197  while (temp_pos.se < this->pos.end.se + additional) {
198  TerrainChunk *chunk = this->get_terrain()->get_chunk(temp_pos);
199 
200  if (chunk == nullptr) {
201  continue;
202  }
203 
204  chunk->get_data(temp_pos.get_pos_on_chunk())->terrain_id = id;
205  temp_pos.se++;
206  }
207  temp_pos.se = this->pos.start.se - additional;
208  temp_pos.ne++;
209  }
210 }

230 bool TerrainObject::operator <(const TerrainObject &other) {
231  if (this == &other) {
232  return false;
233  }
234 
235  auto this_ne = this->pos.draw.ne;
236  auto this_se = this->pos.draw.se;
237  auto other_ne = other.pos.draw.ne;
238  auto other_se = other.pos.draw.se;
239 
240  auto this_ypos = this_ne - this_se;
241  auto other_ypos = other_ne - other_se;
242 
243  if (this_ypos < other_ypos) {
244  return false;
245  }
246  else if (this_ypos == other_ypos) {
247  if (this_ne > other_ne) {
248  return false;
249  }
250  else if (this_ne == other_ne) {
251  return this_se > other_se;
252  }
253  }
254 
255  return true;
256 }

258 void TerrainObject::place_unchecked(const std::shared_ptr<Terrain> &t, coord::phys3 &position) {
259  // storing the position:
260  this->pos = this->get_range(position, *t);
261  this->terrain = t;
262  this->occupied_chunk_count = 0;
263 
264  bool chunk_known = false;
265 
266 
267  // set pointers to this object on each terrain tile
268  // where the building will stand and block the ground
269  for (coord::tile temp_pos : tile_list(this->pos)) {
270  TerrainChunk *chunk = this->get_terrain()->get_chunk(temp_pos);
271 
272  if (chunk == nullptr) {
273  continue;
274  }
275 
276  for (int c = 0; c < this->occupied_chunk_count; c++) {
277  if (this->occupied_chunk[c] == chunk) {
278  chunk_known = true;
279  }
280  }
281 
282  if (not chunk_known) {
283  this->occupied_chunk[this->occupied_chunk_count] = chunk;
284  this->occupied_chunk_count += 1;
285  } else {
286  chunk_known = false;
287  }
288 
289  chunk->get_data(temp_pos.get_pos_on_chunk())->obj.push_back(this);
290  }
291 }

311 coord::phys_t SquareObject::from_edge(const coord::phys3 &point) const {
312  // clamp between start and end
313  coord::phys2 start_phys = this->pos.start.to_phys2();
314  coord::phys2 end_phys = this->pos.end.to_phys2();
315 
316  coord::phys_t cx = std::max(start_phys.ne, std::min(end_phys.ne, point.ne));
317  coord::phys_t cy = std::max(start_phys.se, std::min(end_phys.se, point.se));
318 
319  // distance to clamped point
320  coord::phys_t dx = point.ne - cx;
321  coord::phys_t dy = point.se - cy;
322  return std::hypot(dx, dy);
323 }

348 bool SquareObject::intersects(const TerrainObject &other, const coord::phys3 &position) const {
349  if (const SquareObject *sq = dynamic_cast<const SquareObject *>(&other)) {
350  auto terrain_ptr = this->terrain.lock();
351  if (not terrain_ptr) {
352  throw Error{ERR << "object is not associated to a valid terrain"};
353  }
354 
355  tile_range rng = this->get_range(position, *terrain_ptr);
356  return this->pos.end.ne < rng.start.ne
357  || rng.end.ne < sq->pos.start.ne
358  || rng.end.se < sq->pos.start.se
359  || rng.end.se < sq->pos.start.se;
360  }
361  else if (const RadialObject *rad = dynamic_cast<const RadialObject *>(&other)) {
362  auto terrain_ptr = this->terrain.lock();
363  if (not terrain_ptr) {
364  throw Error{ERR << "object is not associated to a valid terrain"};
365  }
366  // clamp between start and end
367  tile_range rng = this->get_range(position, *terrain_ptr);
368  coord::phys2 start_phys = rng.start.to_phys2();
369  coord::phys2 end_phys = rng.end.to_phys2();
370  coord::phys_t cx = std::max(start_phys.ne, std::min(end_phys.ne, rad->pos.draw.ne));
371  coord::phys_t cy = std::max(start_phys.se, std::min(end_phys.se, rad->pos.draw.se));
372 
373  // distance to square object base
374  coord::phys_t dx = rad->pos.draw.ne - cx;
375  coord::phys_t dy = rad->pos.draw.se - cy;
376  return std::hypot(dx, dy) < rad->phys_radius.to_double();
377  }
378  return false;
379 }

445 std::vector<coord::tile> tile_list(const tile_range &rng) {
446  std::vector<coord::tile> tiles;
447 
448  coord::tile check_pos = rng.start;
449  while (check_pos.ne < rng.end.ne) {
450  while (check_pos.se < rng.end.se) {
451  tiles.push_back(check_pos);
452  check_pos.se += 1;
453  }
454  check_pos.se = rng.start.se;
455  check_pos.ne += 1;
456  }
457 
458  // a case when the objects radius is zero
459  if (tiles.empty()) {
460  tiles.push_back(rng.start);
461  }
462  return tiles;
463  // a case when the objects radius is zero
464  if (tiles.empty()) {
465  tiles.push_back(rng.start);
466  }
467  return tiles;
468 }
