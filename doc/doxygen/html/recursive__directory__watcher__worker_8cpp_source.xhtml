34 QStringList collect_entries_to_watch(const QStringList &root_dirs_paths) {
35  QStringList root_dirs_paths_no_duplicates = root_dirs_paths;
36  root_dirs_paths_no_duplicates.removeDuplicates();
37 
38  QStringList entries_to_watch;
39 
40  std::for_each(std::begin(root_dirs_paths_no_duplicates), std::end(root_dirs_paths_no_duplicates), [&entries_to_watch] (const QString& root_dir_path) {
41  QDirIterator it{root_dir_path, QDirIterator::Subdirectories | QDirIterator::FollowSymlinks};
42 
43  while (it.hasNext()) {
44  entries_to_watch.append(it.next());
45  }
46  });
47 
48  return entries_to_watch;
49 }
