58 void GlWindow::update() {
59  SDL_Event event;
60 
61  while (SDL_PollEvent(&event) != 0) {
62  if (event.type == SDL_WINDOWEVENT) {
63  if (event.window.event == SDL_WINDOWEVENT_RESIZED) {
64  size_t width = event.window.data1;
65  size_t height = event.window.data2;
66  log::log(MSG(dbg) << "Window resized to: " << width << "x" << height);
67  this->size = std::make_pair(width, height);
68  for (auto& cb : this->on_resize) {
69  cb(width, height);
70  }
71  }
72  } else if (event.type == SDL_QUIT) {
73  this->should_be_closed = true;
74  // \todo call on_destroy
75  } else if (event.type == SDL_KEYUP) {
76  auto const ev = *reinterpret_cast<SDL_KeyboardEvent const*>(&event);
77  for (auto& cb : this->on_key) {
78  cb(ev);
79  }
80  // \todo handle keydown
81  } else if (event.type == SDL_MOUSEBUTTONUP) {
82  auto const ev = *reinterpret_cast<SDL_MouseButtonEvent const*>(&event);
83  for (auto& cb : this->on_mouse_button) {
84  cb(ev);
85  }
86  // \todo handle mousedown
87  }
88  }
89 
90  SDL_GL_SwapWindow(this->window);
