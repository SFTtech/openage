11 class Path:
12  """
13  Implements an interface somewhat similar to that of pathlib.Path,
14  but some methods are missing or have different usage, and some new have
15  been added.
16 
17  Represents a specific path in a given FS-Like object; mostly, that
18  object's member methods are simply wrapped.
19 
20  fsobj: fs-like object that is e.g. a cab-archive, a real
21  Directory("/lol"), or anything that is like some filesystem.
22 
23  parts: starting path in the above fsobj,
24  e.g. ["folder", "file"],
25  or "folder/file"
26  or b"folder/file".
27  """
28 
29  # We're re-implementing pathlib.Path's interface, we have no choice about
30  # this. Also, some of these are properties, so the effective count is
31  # lower.
32  # pylint: disable=too-many-public-methods
33 
34  def __init__(self, fsobj, parts=None):
35  if isinstance(parts, str):
36  parts = parts.encode()
37 
38  if isinstance(parts, (bytes, bytearray)):
39  parts = parts.split(b'/')
40 
41  if parts is None:
42  parts = []
43 
44  if not isinstance(parts, (list, tuple)):
45  raise ValueError("path parts must be str, bytes, list or tuple, "
46  "but not: %s" % type(parts))
47 
48  result = []
49  for part in parts:
50  if isinstance(part, str):
51  part = part.encode()
52 
53  if part in (b'.', b''):
54  pass
55  elif part == b'..':
56  try:
57  result.pop()
58  except IndexError:
59  pass
60  else:
61  result.append(part)
62 
63  self.fsobj = fsobj
64 
65  # use tuple instead of list to prevent accidential modification
66  self.parts = tuple(result)

114  elif dmode == "w":

166  """
167  Minimize the path and possibly return a native one.
168  Returns None if there was no native path.
169  """

171  return self.resolve_native_path_r()

176  raise UnsupportedOperation("unsupported resolve mode: " + mode)

178  def resolve_native_path_r(self):
179  """
180  Resolve the path for read access and possibly return
181  a native equivalent.
182  If no native path was found, return None.
183  """
184  resolved_path = self._resolve_r()
185  if resolved_path:
186  # pylint: disable=protected-access
187  return resolved_path._get_native_path()
188  return None
