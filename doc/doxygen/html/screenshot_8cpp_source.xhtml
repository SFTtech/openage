65 bool ScreenshotManager::encode_png(std::shared_ptr<uint8_t> pxdata,
66  coord::viewport_delta size) {
67  std::FILE *fout = NULL;
68  coord::pixel_t width = size.x,
69  height = size.y;
70  auto warn_fn = [] (png_structp /*png_ptr*/, png_const_charp message) {
71  log::log(MSG(err) << "Creating screenshot failed: libpng error: " << message);
72  };
73  auto err_fn = [] (png_structp png_ptr, png_const_charp message) {
74  log::log(MSG(err) << "Creating screenshot failed: libpng error: " << message);
75  longjmp(png_jmpbuf(png_ptr), 1);
76  };
77 
78  png_structp png_ptr = png_create_write_struct(PNG_LIBPNG_VER_STRING,
79  (png_voidp) NULL,
80  err_fn, warn_fn);
81  if (!png_ptr)
82  return false;
83 
84  png_infop info_ptr = png_create_info_struct(png_ptr);
85  if (!info_ptr) {
86  png_destroy_write_struct(&png_ptr, (png_infopp) NULL);
87  return false;
88  }
89 
90  if (setjmp(png_jmpbuf(png_ptr))) {
91  std::fclose(fout);
92  png_destroy_write_struct(&png_ptr, &info_ptr);
93  return false;
94  }
95 
96  std::string filename = this->gen_next_filename();
97  fout = std::fopen(filename.c_str(), "wb");
98  if (fout == NULL) {
99  png_destroy_write_struct(&png_ptr, &info_ptr);
100  log::log(MSG(err) << "Could not open '"<< filename << "': "
101  << std::string(strerror(errno)));
102  return false;
103  }
104 
105  png_init_io(png_ptr, fout);
106 
107  png_set_IHDR(png_ptr, info_ptr, width, height, 8, PNG_COLOR_TYPE_RGB,
108  PNG_INTERLACE_NONE, PNG_COMPRESSION_TYPE_DEFAULT,
109  PNG_FILTER_TYPE_DEFAULT);
110 
111  // Put image row pointer into info_ptr so that we can use the high-level
112  // write interface.
113  std::vector<png_bytep> row_ptrs;
114 
115  // Invert rows.
116  row_ptrs.reserve(height);
117  for (int i = 1; i <= height; i++) {
118  row_ptrs.push_back(pxdata.get() + (height - i) * 4 * width);
119  }
120  png_set_rows(png_ptr, info_ptr, &row_ptrs[0]);
121 
122  //TODO: print ingame message.
123  log::log(MSG(info) << "Saving screenshot to '" << filename << "'.");
124 
125  png_write_png(png_ptr, info_ptr, PNG_TRANSFORM_STRIP_FILLER_AFTER, NULL);
126 
127  std::fclose(fout);
128  png_destroy_write_struct(&png_ptr, &info_ptr);
129  return true;
130 }
