111 void Font::initialize(FT_Library ft_library) {
112  FT_Face ft_face;
113  if (unlikely(FT_New_Face(ft_library, this->description.font_file.c_str(), 0, &ft_face))) {
114  throw Error(MSG(err) << "Failed to create font from " << this->description.font_file);
115  }
116  if (unlikely(FT_Set_Char_Size(ft_face, 0, this->description.size * FREETYPE_UNIT, 72, 72))) {
117  throw Error(MSG(err) << "Failed to set font face size to " << this->description.size);
118  }
119 
120  hb_ft_font_create_referenced(ft_face);
121  // lambda with static_cast to help gcc understand this is ok
122  this->hb_font = hb_ft_font_create(ft_face, [] (void *user_data) -> void {
123  FT_Done_Face(static_cast<FT_FaceRec_ *>(user_data));
124  });
125 }

162 float Font::get_advance_width(const std::string &text) const {
163  hb_buffer_t *buffer = hb_buffer_create();
164  hb_buffer_set_direction(buffer, get_hb_font_direction(this->description));
165  hb_buffer_set_script(buffer, get_hb_font_script(this->description));
166  hb_buffer_set_language(buffer, get_hb_font_language(this->description));
167  hb_buffer_add_utf8(buffer, text.c_str(), text.length(), 0, text.length());
168  hb_shape(this->hb_font, buffer, nullptr, 0);
169 
170  float advance_width = 0.0f;
171 
172  unsigned int glyph_count = 0;
173  hb_glyph_info_t *glyph_info = hb_buffer_get_glyph_infos(buffer, &glyph_count);
174  hb_glyph_position_t *glyph_pos = hb_buffer_get_glyph_positions(buffer, nullptr);
175  for (unsigned int i = 0; i < glyph_count; i++) {
176  advance_width += static_cast<float>(glyph_pos[i].x_advance)/FREETYPE_UNIT;
177 
178  if (i > 0) {
179  codepoint_t glyph = glyph_info[i].codepoint;
180  codepoint_t previous_glyph = glyph_info[i - 1].codepoint;
181  advance_width += this->get_horizontal_kerning(previous_glyph, glyph);
182  }
183  }
184 
185  hb_buffer_destroy(buffer);
186 
187  return advance_width;
188 }

212 std::unique_ptr<unsigned char[]> Font::load_glyph(codepoint_t codepoint, Glyph &glyph) const {
213  FT_Face ft_face = hb_ft_font_get_face(this->hb_font);
214  if (unlikely(FT_Load_Glyph(ft_face, codepoint, FT_LOAD_DEFAULT | FT_LOAD_NO_HINTING | FT_LOAD_RENDER))) {
215  return nullptr;
216  }
217 
218  glyph.codepoint = codepoint;
219  glyph.x_offset = ft_face->glyph->bitmap_left;
220  glyph.y_offset = ft_face->glyph->bitmap_top;
221  glyph.width = ft_face->glyph->bitmap.width;
222  glyph.height = ft_face->glyph->bitmap.rows;
223  glyph.x_advance = static_cast<float>(ft_face->glyph->advance.x)/FREETYPE_UNIT;
224  glyph.y_advance = static_cast<float>(ft_face->glyph->advance.y)/FREETYPE_UNIT;
225 
226  auto glyph_data = std::make_unique<unsigned char[]>(glyph.width * glyph.height);
227  for (size_t i = 0; i < glyph.height; i++) {
228  memcpy(
229  &glyph_data[i * glyph.width],
230  ft_face->glyph->bitmap.buffer + (glyph.height - i - 1) * glyph.width,
231  glyph.width * sizeof(unsigned char)
232  );
233  }
234  return glyph_data;
235 }
