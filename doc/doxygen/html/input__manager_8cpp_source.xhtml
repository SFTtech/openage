66 std::string InputManager::get_bind(const std::string &action_str) {
67  action_t action = this->action_manager->get(action_str);
68  if (this->action_manager->is("UNDEFINED", action)) {
69  return "";
70  }
71 
72  auto it = this->keys.find(action);
73  if (it == this->keys.end()) {
74  return " ";
75  }
76 
77  switch (it->second.cc.eclass) {
78  case event_class::MOUSE_BUTTON :
79  return this->mouse_bind_to_string(it->second);
80  case event_class::MOUSE_WHEEL :
81  return this->wheel_bind_to_string(it->second);
82  default:
83  return this->key_bind_to_string(it->second);
84  }
85  return "";
86 }

92  return false;

94 
95  Event ev = text_to_event(bind_str);

103  return true;

188 bool InputManager::trigger(const Event &e) {
189  if (this->ignored(e)) {
190  return false;
191  }
192 
193  // arg passed to receivers
194  action_arg_t arg{e, this->mouse_position, this->mouse_motion, {}};
195 
196  for (auto &it : this->keys) {
197  if (e == it.second) {
198  arg.hints.emplace_back(it.first);
199  }
200  }
201 
202  // Check context list on top of the stack (most recent bound first)
203  for (auto it = this->contexts.rbegin(); it != this->contexts.rend(); ++it) {
204  if ((*it)->execute_if_bound(arg)) {
205  return true;
206  }
207  }
208 
209  // If no local keybinds were bound, check the global keybinds
210  return this->global_context.execute_if_bound(arg);
211 }

214 void InputManager::set_state(const Event &ev, bool is_down) {
215  if (this->ignored(ev)) {
216  return;
217  }
218 
219  // update key states
220  this->keymod = ev.mod;
221  bool was_down = this->states[ev.cc];
222  this->states[ev.cc] = is_down;
223 
224  // a key going from pressed to unpressed
225  // will automatically trigger event handling
226  if (was_down && !is_down) {
227  this->trigger(ev);

260 bool InputManager::is_mouse_at_edge(Edge edge, int window_size) {
261  int x, y;
262  SDL_GetMouseState(&x, &y);
263 
264  // Border width to consider screen edges for scrolling.
265  // AoE II appears to be approx 10px.
266  // \todo make configurable through cvar.
267  const int edge_offset = 10;
268 
269  if (edge == Edge::LEFT && x <= edge_offset) {
270  return true;
271  }
272  if (edge == Edge::RIGHT && x >= window_size - edge_offset - 1) {
273  return true;
274  }
275  if (edge == Edge::UP && y <= edge_offset) {
276  return true;
277  }
278  if (edge == Edge::DOWN && y >= window_size - edge_offset - 1) {
279  return true;
280  }
281 
282  return false;
283 }

315 bool InputManager::on_input(SDL_Event *e) {
316 
317  // top level input handler
318  switch (e->type) {
319 
320  case SDL_KEYUP: {
321  SDL_Keycode code = reinterpret_cast<SDL_KeyboardEvent *>(e)->keysym.sym;
322  Event ev = sdl_key(code, SDL_GetModState());
323  this->set_state(ev, false);
324  break;
325  } // case SDL_KEYUP
326 
327  case SDL_KEYDOWN: {
328  SDL_Keycode code = reinterpret_cast<SDL_KeyboardEvent *>(e)->keysym.sym;
329  this->set_state(sdl_key(code, SDL_GetModState()), true);
330  break;
331  } // case SDL_KEYDOWN
332 
333  case SDL_TEXTINPUT: {
334  this->trigger(utf8(e->text.text));
335  break;
336  } // case SDL_TEXTINPUT
337 
338  case SDL_MOUSEBUTTONUP: {
339  this->set_relative(false);
340  this->trigger(sdl_mouse_up_down(e->button.button, true, SDL_GetModState()));
341  Event ev = sdl_mouse(e->button.button, SDL_GetModState());
342  this->set_state(ev, false);
343  break;
344  } // case SDL_MOUSEBUTTONUP
345 
346  case SDL_MOUSEBUTTONDOWN: {
347 
348  // \todo set which buttons
349  if (e->button.button == 2) {
350  this->set_relative(true);
351  }
352  this->trigger(sdl_mouse_up_down(e->button.button, false, SDL_GetModState()));
353  Event ev = sdl_mouse(e->button.button, SDL_GetModState());
354  this->set_state(ev, true);
355  break;
356  } // case SDL_MOUSEBUTTONDOWN
357 
358  case SDL_MOUSEMOTION: {
359  if (this->relative_mode) {
360  this->set_motion(e->motion.xrel, e->motion.yrel);
361  }
362  else {
363  this->set_mouse(e->button.x, e->button.y);
364  }
365 
366  // must occur after setting mouse position
367  Event ev(event_class::MOUSE_MOTION, 0, this->get_mod());
368  this->trigger(ev);
369  break;
370  } // case SDL_MOUSEMOTION
371 
372  case SDL_MOUSEWHEEL: {
373  Event ev = sdl_wheel(e->wheel.y, SDL_GetModState());
374  this->trigger(ev);
375  break;
376  } // case SDL_MOUSEWHEEL
377 
378  } // switch (e->type)
379 
380  return true;
381 }
