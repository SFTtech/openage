15  Wrapper file-like object that adds seek functionality to a read-only,
16  unseekable stream.
17 
18  For this purpose, all data read from that stream is cached.

53  def read(self, size=-1):
54  if size < 0:
55  size = INF
56 
57  # see if we have already discarded the requested data
58  if self.buf.hasbeendiscarded(self.pos):
59  # shit.
60  # try to reset the stream and start over, hoping that the stream
61  # actually implements .reset().
62  self.wrapped.reset()
