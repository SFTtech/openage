148 void Player::active_unit_added(Unit *unit, bool from_pending) {
149  // check if unit is actually active
150  if (this->is_unit_pending(unit)) {
151  this->units_pending[unit->unit_type->id()] += 1;
152  return;
153  }
154 
155  if (from_pending) {
156  this->units_pending[unit->unit_type->id()] -= 1;
157  }
158 
159  this->units_have[unit->unit_type->id()] += 1;
160  this->units_had[unit->unit_type->id()] += 1;
161  // \todo handle here building dependencies
162 
163  // population
164  if (unit->has_attribute(attr_type::population)) {
165  auto popul = unit->get_attribute<attr_type::population>();
166  if (popul.demand > 0) {
167  this->population.demand_population(popul.demand);
168  }
169  if (popul.capacity > 0) {
170  this->population.add_capacity(popul.capacity);
171  }
172  }
173 
174  // resources capacity
175  if (unit->has_attribute(attr_type::storage)) {
176  auto storage = unit->get_attribute<attr_type::storage>();
177  this->resources_capacity += storage.capacity;
178  this->on_resources_change();
179  }
180 
181  // score
182  // \todo improve selectors
183  if (unit->unit_type->id() == 82 || unit->unit_type->id() == 276) { // Castle, Wonder
184  this->score.add_score(score_category::society, unit->unit_type->cost.get(*this).sum() * 0.2);
185  } else if (unit->has_attribute(attr_type::building) || unit->has_attribute(attr_type::population)) { // building, living
186  this->score.add_score(score_category::economy, unit->unit_type->cost.get(*this).sum() * 0.2);
187  }
188 
189  // \todo handle here on create unit triggers
190  // \todo check for unit based win conditions
191 }

201  // \todo handle here building dependencies

206  if (popul.demand > 0) {

217  this->resources_capacity -= storage.capacity;
218  this->on_resources_change();

220
