122 class CaseIgnoringDirectory(Directory):
123  """
124  Like directory, but all given paths must be lower-case,
125  and will be resolved to the actual correct case.
126 
127  The one exception is the constructor argument:
128  It _must_ be in the correct case.
129  """

131  super().__init__(path, create_if_missing)
132  self.cache = {(): ()}

136  return "Directory({})".format(self.path.decode(errors='replace'))

138  def actual_name(self, stem, name):
139  """
140  If the (lower-case) path that's given in stem exists,
141  fetches the actual name for the given lower-case name.
142  """
143  try:
144  listing = self.listings[tuple(stem)]
145  except KeyError:
146  # the directory has not been listed yet.
147  try:
148  filelist = os.listdir(os.path.join(self.path, *stem))
149  except FileNotFoundError:
150  filelist = []
151 
152  listing = {}
153  for filename in filelist:
154  if filename.lower() != filename:
155  listing[filename.lower()] = filename
156  self.listings[tuple(stem)] = listing
157 
158  try:
159  return listing[name]
160  except KeyError:
161  return name

163  def resolve(self, parts):
164  parts = [part.lower() for part in parts]
165 
166  i = 0
167  for i in range(len(parts), -1, -1):
168  try:
169  result = list(self.cache[tuple(parts[:i])])
170  break
171  except KeyError:
172  pass
173  else:
174  raise RuntimeError("code flow error")
175 
176  # result now contains the case-corrected path for parts[:i].
177  # we need to append the path for parts[i:].
178  for part in parts[i:]:
179  result.append(self.actual_name(result, part))
180  self.cache[tuple(parts[:len(result)])] = tuple(result)
181 
182  return os.path.join(self.path, *result)
