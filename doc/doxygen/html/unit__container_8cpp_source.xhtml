143 bool UnitContainer::update_all(time_nsec_t lastframe_duration) {
144  // update everything and find objects with no actions
145  std::vector<id_t> to_remove;
146 
147  for (auto &obj : this->live_units) {
148  obj.second->update(lastframe_duration);
149 
150  if (not obj.second->has_action()) {
151  to_remove.push_back(obj.first);
152  }
153  }
154 
155  // cleanup and removal of objects
156  for (auto &obj : to_remove) {
157 
158  // unique pointer triggers cleanup
159  this->live_units.erase(obj);
160  }
161  return true;
