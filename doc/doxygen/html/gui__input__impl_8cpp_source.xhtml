59 int sdl_mouse_mask_to_qt(Uint32 state) {
60  return (state & (Qt::LeftButton | Qt::XButton1 | Qt::XButton2)) | ((state & Qt::RightButton) << 1) | ((state & Qt::MiddleButton) >> 1);

88 bool GuiInputImpl::process(SDL_Event *e) {
89 
90  switch (e->type) {
91  case SDL_MOUSEMOTION: {
92  QMouseEvent ev{QEvent::MouseMove, QPoint{e->motion.x, e->motion.y}, Qt::MouseButton::NoButton, this->mouse_buttons_state = sdl_mouse_state_to_qt(e->motion.state), Qt::KeyboardModifier::NoModifier};
93  ev.setAccepted(false);
94 
95  // Allow dragging stuff under the gui overlay.
96  return relay_input_event(&ev, e->motion.state & (SDL_BUTTON_LMASK | SDL_BUTTON_MMASK | SDL_BUTTON_RMASK));
97  break;
98  }
99 
100  case SDL_MOUSEBUTTONDOWN: {
101  auto button = sdl_mouse_btn_to_qt(e->button.button);
102  QMouseEvent ev{QEvent::MouseButtonPress, QPoint{e->button.x, e->button.y}, button, this->mouse_buttons_state |= button, Qt::KeyboardModifier::NoModifier};
103  ev.setAccepted(false);
104 
105  bool accepted = relay_input_event(&ev);
106 
107  if (e->button.clicks == 2) {
108  QMouseEvent ev_dbl{QEvent::MouseButtonDblClick, QPoint{e->button.x, e->button.y}, button, this->mouse_buttons_state, Qt::KeyboardModifier::NoModifier};
109  ev_dbl.setAccepted(false);
110  accepted = relay_input_event(&ev_dbl) || accepted;
111  }
112 
113  return accepted;
114  break;
115  }
116 
117  case SDL_MOUSEBUTTONUP: {
118  auto button = sdl_mouse_btn_to_qt(e->button.button);
119  QMouseEvent ev{QEvent::MouseButtonRelease, QPoint{e->button.x, e->button.y}, button, this->mouse_buttons_state &= ~button, Qt::KeyboardModifier::NoModifier};
120  ev.setAccepted(false);
121 
122  // Allow dragging stuff under the gui overlay: when no item is grabbed, it probably means that initial MousButtonPress was outside gui.
123  return relay_input_event(&ev, true);
124  break;
125  }
126 
127  case SDL_MOUSEWHEEL: {
128  QPoint pos;
129  SDL_GetMouseState(&pos.rx(), &pos.ry());
130 
131  QWheelEvent ev{pos, pos, QPoint{}, QPoint{e->wheel.x, e->wheel.y}, 0, Qt::Orientation{}, this->mouse_buttons_state, Qt::KeyboardModifier::NoModifier};
132  ev.setAccepted(false);
133 
134  return relay_input_event(&ev);
135  break;
136  }
137 
138  case SDL_KEYDOWN: {
139  QKeyEvent ev{QEvent::KeyPress, sdl_key_to_qt(e->key.keysym.sym), Qt::NoModifier, QChar((short) e->key.keysym.sym)};
140  ev.setAccepted(false);
141  return relay_input_event(&ev);
142  break;
143  }
144 
145  case SDL_KEYUP: {
146  QKeyEvent ev{QEvent::KeyRelease, sdl_key_to_qt(e->key.keysym.sym), Qt::NoModifier, QChar((short) e->key.keysym.sym)};
147  ev.setAccepted(false);
148  return relay_input_event(&ev);
149  break;
150  }
151 
152  default:
153  break;
154  }
155 
156  return false;
157 }

159 bool GuiInputImpl::relay_input_event(QEvent *ev, bool only_if_grabbed) {
160  const bool logic_diff_input = this->game_logic_updater->get_thread() != QThread::currentThread();
161  std::atomic<bool> processed{false};
162 
163  emit this->input_event(&processed, ev, only_if_grabbed);
164 
165  // We have sent an event to the gui thread and want a response about collision. But the gui can be
166  // executing a getter that blocks the gui thread while doing something in the logic thread.
167  // So, if the logic thread is the same as the input thread, it should be running somehow.
168  //
169  // TODO: if/when the logic thread or input thread of the main game is made separate, give mutex or
170  // queue to the gui in order to replace this busy wait.
171  if (!logic_diff_input)
172  while (!processed) {
173  this->game_logic_updater->process_callbacks();
174  QThread::usleep(1);
175  }
176 
177  return ev->isAccepted();
178 }
