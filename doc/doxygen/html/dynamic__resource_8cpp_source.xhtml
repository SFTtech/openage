73 audio_chunk_t DynamicResource::get_data(size_t position, size_t data_length) {
74  size_t resource_chunk_index = position / this->chunk_size;
75  size_t chunk_offset = position % this->chunk_size;
76 
77  // check if the audio chunk is currently loading or already ready
78  auto chunk_map_it = this->chunks.find(resource_chunk_index);
79  if (chunk_map_it != std::end(this->chunks)) {
80  auto chunk_info = chunk_map_it->second;
81  int16_t *data = chunk_info->buffer.get() + chunk_offset;
82 
83  switch (chunk_info->state.load()) {
84  case chunk_info_t::state_t::EMPTY:
85  this->chunks.erase(chunk_map_it);
86  // signal end of stream
87  return {nullptr, 0};
88 
89  case chunk_info_t::state_t::LOADING:
90  // signal that resource is not ready yet
91  return {nullptr, 1};
92 
93  case chunk_info_t::state_t::READY:
94  // start to preload the next chunks!
95  this->start_preloading(resource_chunk_index);
96 
97  // calculate actual data length
98  if (chunk_info->size - chunk_offset >= data_length) {
99  return {data, data_length};
100  } else {
101  return {data, chunk_info->size - chunk_offset};
102  }
103  }
104  }
105 
106  // guard the .empty(), .front() and .pop() calls
107  auto chunk_info_lock = this->decay_queue.lock();
108 
109  // we don't have next data to load!
110  if (this->decay_queue.empty()) {
111  throw Error{ERR << "chunk decay queue ran dry, resource is dying."};
112  }
113 
114  // obtain a chunk info element to be used and overwritten
115  auto chunk_info = this->decay_queue.front();
116  this->decay_queue.pop();
117 
118  chunk_info_lock.unlock();
119 
120  // map it to the corresponding resource offset
121  this->chunks.insert({resource_chunk_index, chunk_info});
122 
123  // start a background job to get the data of that chunk
124  size_t resource_chunk_offset = resource_chunk_index * this->chunk_size;
125  this->load_chunk_async(chunk_info, resource_chunk_offset);
126 
127  // and also load more chunks starting at the current chunk index
128  this->start_preloading(resource_chunk_index);
129 
130  return {nullptr, 1};
131 }

134 void DynamicResource::start_preloading(size_t resource_chunk_index) {
135  size_t resource_chunk_offset = resource_chunk_index * this->chunk_size;
136 
137  // from the current index, enqueue the load of `threshold` more chunks
138  for (int i = 1; i < this->preload_amount; i++) {
139  resource_chunk_index += 1;
140  resource_chunk_offset += this->chunk_size;
141 
142  auto chunk_map_it = this->chunks.find(resource_chunk_index);
143 
144  // the chunk is either in loading state or ready
145  // we do this to no longer preload if the stream is at its end.
146  if (chunk_map_it != std::end(this->chunks)) {
147  auto chunk_info = chunk_map_it->second;
148  if (chunk_info->state.load() == chunk_info_t::state_t::EMPTY) {
149  this->chunks.erase(chunk_map_it);
150 
151  // we encountered the end of stream!
152  break;
153  }
154  // else, the chunk is known already and loading or even loaded,
155  // no need to trigger the load.
156  }
157  else {
158  // chunk is unknown yet, so let's load it.
159 
160  auto decay_queue_lock = this->decay_queue.lock();
161  // obtain the least-recently-used chunk to overwrite.
162  auto local_chunk_info = this->decay_queue.front();
163  this->decay_queue.pop();
164  decay_queue_lock.unlock();
165 
166  this->chunks.insert({resource_chunk_index, local_chunk_info});
167  this->load_chunk_async(local_chunk_info, resource_chunk_offset);
168  }
169  }
170 }

173 void DynamicResource::load_chunk_async(std::shared_ptr<chunk_info_t> chunk_info,
174  size_t resource_chunk_offset) {
175  chunk_info->state.store(chunk_info_t::state_t::LOADING);
176 
177  // the `chunk_info` will stay out of the `decay_queue`
178  // until it was loaded. So it's possible the queue runs dry.
179 
180  ENSURE(this->loader, "tried to load chunk without audio loader!");
181 
182  auto loading_function = [this,chunk_info,resource_chunk_offset]() -> int {
183  int16_t *buffer = chunk_info->buffer.get();
184  size_t loaded = this->loader->load_chunk(buffer, resource_chunk_offset, this->chunk_size);
185  if (loaded == 0) {
186  chunk_info->state.store(chunk_info_t::state_t::EMPTY);
187  this->decay_queue.push(chunk_info);
188  } else {
189  chunk_info->state.store(chunk_info_t::state_t::READY);
190  chunk_info->size = loaded;
191  this->decay_queue.push(chunk_info);
192  }
193  // as the job manager currently does not support executing void
194  // functions, we return zero
195  return 0;
196  };
197  this->loading_job_group.enqueue<int>(loading_function);
198 }
