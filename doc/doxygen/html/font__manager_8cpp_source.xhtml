14 std::string FontManager::get_font_filename(const char *family, const char *style) {
15  // Initialize fontconfig
16  if (!FcInit()) {
17  throw Error{ERR << "Failed to initialize fontconfig."};
18  }
19 
20  FcPattern *font_pattern = FcPatternBuild(nullptr, FC_FAMILY, FcTypeString, family, nullptr);
21  FcPatternBuild(font_pattern, FC_STYLE, FcTypeString, style, nullptr);
22 
23  FcChar8 *query_string = FcNameUnparse(font_pattern);
24  log::log(DBG << "Font queried: " << query_string);

26

30 
31  // get attibute FC_FILE (= filename) of best-matched font
32  FcChar8 *font_filename_tmp;
33  if (FcPatternGetString(font_match, FC_FILE, 0, &font_filename_tmp) != FcResultMatch) {
34  throw Error(ERR << "Fontconfig could not provide font " << family << " " << style);
35  }
36 
37  std::string font_filename{reinterpret_cast<char *>(font_filename_tmp)};
38 
39  log::log(DBG << "Font file: " << font_filename);
40 
41  // deinitialize fontconfig.
42  FcPatternDestroy(font_match);
43  FcPatternDestroy(font_pattern);
44  FcFini();
45

47 }
